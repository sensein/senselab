senselab.audio.tasks.classification.speech_emotion_recognition.api

Speech emotion recognition API.

Supports discrete (categorical) and continuous (dimensional) SER models. Continuous SER models that have broken config.json fields (e.g., vocab_size: null) are run in an isolated subprocess venv with pinned huggingface-hub<1.0.

 41# Subprocess venv for continuous SER models with broken configs
 42_CONT_SER_VENV = "continuous-ser"
 43_CONT_SER_REQUIREMENTS = [
 44    "transformers>=4.40,<5",
 45    "huggingface-hub>=0.34,<1.0",  # 1.x strict validation breaks some model configs
 46    "torch>=2.8",
 47    "soundfile",
 48    "numpy",
 49    "safetensors",
 50]

 53_CONT_SER_WORKER = r"""
 54import json
 55import sys
 56from pathlib import Path
 57
 58import numpy as np
 59import soundfile as sf
 60import torch
 61from transformers import AutoConfig, pipeline
 62
 63args = json.loads(sys.stdin.read())
 64audio_paths = args["audio_paths"]
 65model_id = args["model_id"]
 66revision = args["revision"]
 67device = args["device"]
 68output_dir = args["output_dir"]
 69
 70pipe = pipeline(
 71    task="audio-classification",
 72    model=model_id,
 73    revision=revision,
 74    device=device,
 75    function_to_apply="none",
 76)
 77
 78results = []
 79for audio_path in audio_paths:
 80    data, sr = sf.read(audio_path, dtype="float32")

 88def classify_emotions_from_speech(
 89    audios: List[Audio],
 90    model: SenselabModel,
 91    device: Optional[DeviceType] = None,
 92    **kwargs: Any,  # noqa: ANN401
 93) -> List[AudioClassificationResult]:
 94    """Classify all audios using the given speech emotion recognition model.
 95
 96    Args:
 97        audios: The list of audio objects to be classified.
 98        model: The model used for classification, should be trained for recognizing emotions.
 99        device: The device to run the model on (default is None).
100        **kwargs: Additional keyword arguments to pass to the classification function.
101
102    Returns:
103        List of speech emotion recognition results.
104    """
105    if isinstance(model, SpeechBrainModel):
106        return _classify_speechbrain_ser(audios, model, device)
107
108    if isinstance(model, HFModel):
109        model_info = model.get_model_info()
110        tags = model_info.tags or []
111
112        ser_type = _get_ser_type(model)
113
114        if (
115            not ("speech-emotion-recognition" in tags or "emotion-recognition" in tags)
116            and ser_type == SERType.NOT_RECOGNIZED
117        ):
118            raise ValueError(
119                f"The model '{model.path_or_uri}' is not suitable for speech emotion recognition. Please "
120                "validate that it has the correct tags or use the more generic "
121                "'audio_classification_with_hf_models' function."
122            )
123
124        if ser_type == SERType.CONTINUOUS:
125            return _classify_continuous_ser_venv(audios, model, device)
126        if ser_type == SERType.CONTINUOUS:
127            output_function_to_apply = kwargs.get("function_to_apply", None)
128            if output_function_to_apply:
129                if output_function_to_apply != "none":
130                    warnings.warn(
131                        "Senselab predicts that you are using a continuous SER model but have "
132                        "specified the parameter `function_to_apply` as something other than none. This "
133                        "might create side effects when dealing with continuous values that do not "
134                        "necessarily represent probabilities."
135                    )
136            else:
137                kwargs["function_to_apply"] = "none"
138        return HuggingFaceAudioClassifier.classify_audios_with_transformers(
139            audios=audios, model=model, device=device, **kwargs
140        )
141
142    raise NotImplementedError(
143        "Only Hugging Face and SpeechBrain models are supported. Pass an HFModel or SpeechBrainModel instance."
144    )

147def _classify_continuous_ser_venv(
148    audios: List[Audio],
149    model: HFModel,
150    device: Optional[DeviceType] = None,
151) -> List[AudioClassificationResult]:
152    """Run continuous SER in an isolated subprocess venv.
153
154    Some continuous SER models have broken config.json (e.g., vocab_size: null)
155    that huggingface-hub >=1.0 strict validation rejects. This runs them in a
156    venv with pinned huggingface-hub<1.0.
157    """
158    device_type, _ = _select_device_and_dtype(
159        user_preference=device, compatible_devices=[DeviceType.CUDA, DeviceType.CPU]
160    )
161
162    venv_dir = ensure_venv(_CONT_SER_VENV, _CONT_SER_REQUIREMENTS, python_version=_CONT_SER_PYTHON)
163    python = str(venv_dir / "bin" / "python")
164
165    with tempfile.TemporaryDirectory(prefix="senselab-ser-") as tmpdir:
166        tmp = Path(tmpdir)
167
168        audio_paths = []
169        for i, audio in enumerate(audios):
170            path = str(tmp / f"audio_{i}.flac")
171            audio.save_to_file(path, format="flac")
172            audio_paths.append(path)
173
174        input_json = json.dumps(
175            {
176                "audio_paths": audio_paths,
177                "model_id": str(model.path_or_uri),
178                "revision": model.revision or "main",
179                "device": device_type.value,
180                "output_dir": str(tmp),
181            }
182        )
183
184        result = subprocess.run(
185            [python, "-c", _CONT_SER_WORKER],
186            input=input_json,
187            capture_output=True,
188            text=True,
189            timeout=600,
190        )
191
192        output = parse_subprocess_result(result, "Continuous SER")
193
194        results = []
195        for classification_list in output.get("results", []):
196            labels = [c["label"] for c in classification_list]
197            scores = [float(c["score"]) for c in classification_list]
198            results.append(AudioClassificationResult(labels=labels, scores=scores))
199
200        return results

203def _get_ser_type(model: HFModel) -> SERType:
204    """Get the type of SER the model is likely used for based on the labels it is set to predict."""
205    try:
206        from transformers import AutoConfig
207
208        config = AutoConfig.from_pretrained(model.path_or_uri, revision=model.revision)
209    except Exception:
210        # Fall back to raw config dict for models with invalid fields
211        from huggingface_hub import hf_hub_download
212
213        config_path = hf_hub_download(model.path_or_uri, "config.json", revision=model.revision)
214        with open(config_path) as f:
215            config_dict = json.load(f)
216        config = SimpleNamespace(id2label=config_dict.get("id2label", {}))
217    id2label = config.id2label
218    if id2label:
219        labels = list(id2label.values())
220        if "positive" in labels and "negative" in labels and "neutral" in labels:
221            return SERType.VALENCE
222        if "valence" in labels or "arousal" in labels or "dominance" in labels:
223            return SERType.CONTINUOUS
224        if (
225            "happy" in labels
226            or "happiness" in labels
227            or "joy" in labels
228            or "angry" in labels
229            or "anger" in labels
230            or "sad" in labels
231            or "sadness" in labels
232            or "fear" in labels
233            or "disgust" in labels
234            or "surprise" in labels
235        ):
236            return SERType.DISCRETE
237    return SERType.NOT_RECOGNIZED

248def _classify_speechbrain_ser(
249    audios: List[Audio],
250    model: SpeechBrainModel,
251    device: Optional[DeviceType] = None,
252) -> List[AudioClassificationResult]:
253    """Classify emotions using a SpeechBrain model.
254
255    SpeechBrain SER models (e.g., ``speechbrain/emotion-recognition-wav2vec2-IEMOCAP``)
256    define custom module names (``wav2vec2``, ``avg_pool``, ``output_mlp``) that
257    don't match the generic ``EncoderClassifier`` expectations. This function
258    loads them via a lightweight ``Pretrained`` subclass.
259
260    Args:
261        audios: Audio objects (must be mono, 16 kHz).
262        model: A SpeechBrainModel pointing to a HuggingFace repo.
263        device: Device to run on.
264
265    Returns:
266        List of AudioClassificationResult, one per audio.
267    """
268    from speechbrain.inference import Pretrained
269
270    device_type, _ = _select_device_and_dtype(
271        user_preference=device, compatible_devices=[DeviceType.CUDA, DeviceType.CPU]
272    )
273
274    key = f"{model.path_or_uri}-{model.revision or 'main'}-{device_type.value}"
275    if key not in _speechbrain_ser_models:
276        # Discover required modules from the model's hyperparams
277        _speechbrain_ser_models[key] = _load_speechbrain_ser_model(model, device_type)
278
279    recognizer, label_list = _speechbrain_ser_models[key]
280
281    if not audios:
282        return []
283
284    results: List[AudioClassificationResult] = []
285    for audio in audios:
286        if audio.waveform.shape[0] != 1:
287            raise ValueError("Only mono audio is supported for SpeechBrain SER.")
288
289        wavs = audio.waveform.to(recognizer.device).float()
290        wav_lens = torch.ones(wavs.shape[0], device=recognizer.device)
291
292        with torch.no_grad():
293            # Walk through the model's modules in order
294            out = wavs
295            for mod_name in recognizer.MODULES_NEEDED:
296                mod = getattr(recognizer.mods, mod_name)
297                if mod_name == recognizer.MODULES_NEEDED[0]:
298                    # First module (feature extractor) gets wavs + lens
299                    out = mod(wavs, wav_lens)
300                elif "pool" in mod_name:
301                    out = mod(out, wav_lens)
302                    out = out.view(out.shape[0], -1)
303                else:
304                    out = mod(out)
305
306            probs = torch.nn.functional.softmax(out, dim=-1)
307
308        prob_list = probs[0].cpu().tolist()
309        results.append(AudioClassificationResult(labels=label_list, scores=prob_list))
310
311    return results

314def _load_speechbrain_ser_model(model: SpeechBrainModel, device_type: DeviceType) -> tuple:
315    """Load a SpeechBrain SER model and discover its structure.
316
317    Returns:
318        Tuple of (recognizer, label_list).
319    """
320    import yaml  # type: ignore[import-untyped]
321    from huggingface_hub import hf_hub_download
322    from speechbrain.inference import Pretrained
323
324    # Download hyperparams to discover MODULES_NEEDED
325    hp_path = hf_hub_download(str(model.path_or_uri), "hyperparams.yaml", revision=model.revision)
326
327    # SpeechBrain YAML uses custom tags (!new:, !ref, etc.) that yaml.safe_load
328    # cannot handle. Use a permissive loader subclass that ignores unknown tags.
329    class _PermissiveLoader(yaml.SafeLoader):
330        pass
331
332    _PermissiveLoader.add_multi_constructor("", lambda loader, suffix, node: None)
333    with open(hp_path) as f:
334        hparams = yaml.load(f, Loader=_PermissiveLoader)  # noqa: S506
335
336    modules_needed = hparams.get("MODULES_NEEDED") if hparams else None
337
338    if not modules_needed:
339        raise ValueError(
340            f"Could not determine MODULES_NEEDED from {model.path_or_uri}/hyperparams.yaml. "
341            "This SpeechBrain model may need a custom inference class."
342        )
343
344    # Create a dynamic Pretrained subclass with the correct modules
345    recognizer_cls = type(
346        "SpeechBrainSER",
347        (Pretrained,),
348        {"MODULES_NEEDED": modules_needed},
349    )
350
351    run_opts = {"device": device_type.value}
352    recognizer = recognizer_cls.from_hparams(  # type: ignore[attr-defined]
353        source=str(model.path_or_uri),
354        run_opts=run_opts,
355    )
356
357    # Extract label names from label_encoder if available
358    label_list: List[str] = []
359    if hasattr(recognizer.hparams, "label_encoder"):
360        le = recognizer.hparams.label_encoder
361        if hasattr(le, "ind2lab"):
362            n_classes = len(le.ind2lab)
363            label_list = [le.ind2lab.get(i, f"class_{i}") for i in range(n_classes)]
364
365    if not label_list:
366        logger.warning("Could not extract label names from SpeechBrain model; using generic indices.")
367
368    return recognizer, label_list
