senselab.utils.compatibility_test_runner

Compatibility boundary tester.

Probes version boundaries declared in the compatibility matrix to verify they are accurate. For each function × version combination:

  1. Within range: verify the function works (import + basic call)
  2. Outside range: verify the function indeed fails

Results are written to a JSON report that can be used to update the matrix ranges — widening them when functions pass on untested versions, or tightening when they fail within the declared range.

This is NOT part of the regular test suite. Run explicitly:

uv run python -m senselab.utils.compatibility_test_runner [--update]
Or via pytest with the compat marker:

uv run pytest -m compat --no-header

 37@dataclass
 38class ProbeResult:
 39    """Result of probing a single function × version combination."""
 40
 41    function_key: str
 42    python_version: str
 43    torch_version: str
 44    expected_in_range: bool
 45    actual_passed: bool
 46    error: Optional[str] = None
 47
 48    @property
 49    def boundary_correct(self) -> bool:
 50        """True if the matrix boundary matches reality."""
 51        return self.expected_in_range == self.actual_passed
 52
 53    @property
 54    def should_widen(self) -> bool:
 55        """True if the matrix says out-of-range but function actually works."""
 56        return not self.expected_in_range and self.actual_passed
 57
 58    @property
 59    def should_tighten(self) -> bool:
 60        """True if the matrix says in-range but function actually fails."""
 61        return self.expected_in_range and not self.actual_passed

 64@dataclass
 65class CompatibilityReport:
 66    """Full report from a compatibility probe run."""
 67
 68    timestamp: str = field(default_factory=lambda: datetime.now(timezone.utc).isoformat())
 69    host_python: str = field(default_factory=lambda: f"{sys.version_info.major}.{sys.version_info.minor}")
 70    results: list[ProbeResult] = field(default_factory=list)
 71
 72    @property
 73    def boundaries_correct(self) -> list[ProbeResult]:
 74        """Results where the matrix boundary matches reality."""
 75        return [r for r in self.results if r.boundary_correct]
 76
 77    @property
 78    def should_widen(self) -> list[ProbeResult]:
 79        """Functions that work outside their declared range."""
 80        return [r for r in self.results if r.should_widen]
 81
 82    @property
 83    def should_tighten(self) -> list[ProbeResult]:
 84        """Functions that fail within their declared range."""
 85        return [r for r in self.results if r.should_tighten]
 86
 87    def to_json(self) -> str:
 88        """Serialize to JSON."""
 89        return json.dumps(asdict(self), indent=2)
 90
 91    def summary(self) -> str:
 92        """Human-readable summary."""
 93        lines = [
 94            f"Compatibility Probe Report ({self.timestamp})",
 95            f"  Host Python: {self.host_python}",
 96            f"  Total probes: {len(self.results)}",
 97            f"  Boundaries correct: {len(self.boundaries_correct)}",
 98            f"  Should widen (works outside range): {len(self.should_widen)}",
 99            f"  Should tighten (fails inside range): {len(self.should_tighten)}",
100        ]
101        if self.should_widen:
102            lines.append("\n  Candidates to WIDEN:")
103            for r in self.should_widen:
104                lines.append(f"    {r.function_key}: passed on Python {r.python_version} / torch {r.torch_version}")
105        if self.should_tighten:
106            lines.append("\n  Candidates to TIGHTEN:")
107            for r in self.should_tighten:
108                lines.append(
109                    f"    {r.function_key}: FAILED on Python {r.python_version} / torch {r.torch_version}: {r.error}"
110                )
111        return "\n".join(lines)

114def _probe_function(
115    function_key: str,
116    python_version: str,
117    torch_version: str,
118) -> ProbeResult:
119    """Probe whether a function's imports work on a specific version combo.
120
121    Creates a temporary venv with the specified Python + torch version
122    and tries to import the function's dependencies.
123    """
124    entry = COMPATIBILITY_MATRIX[function_key]
125
126    expected_py = entry.python_versions.contains(f"{python_version}.0")
127    expected_torch = entry.torch_versions.contains(f"{torch_version}.0")
128    expected_in_range = expected_py and expected_torch
129
130    if entry.isolated:
131        # For isolated backends, just check the version range — actual
132        # testing happens in the subprocess venv
133        return ProbeResult(
134            function_key=function_key,
135            python_version=python_version,
136            torch_version=torch_version,
137            expected_in_range=expected_in_range,
138            actual_passed=expected_in_range,  # trust the range for isolated
139        )
140
141    # Build a minimal test script that imports the deps
142    dep_imports = []
143    import_map = {
144        "pyannote-audio": "pyannote.audio",
145        "coqui-tts": "TTS",
146        "opencv-python-headless": "cv2",
147        "sentence-transformers": "sentence_transformers",
148    }
149    for dep in entry.required_deps:
150        mod = import_map.get(dep, dep.replace("-", "_"))
151        dep_imports.append(f"import {mod}")
152
153    test_script = "\n".join(
154        [
155            "import sys",
156            *dep_imports,
157            "print('OK')",
158        ]
159    )
160
161    # Create a probe venv
162    venv_name = f"probe-py{python_version}-torch{torch_version}"
163    reqs = [f"torch=={torch_version}.*"]
164    for dep in entry.required_deps:
165        ver_spec = entry.dep_versions.get(dep, "")
166        reqs.append(f"{dep}{ver_spec}")
167
168    try:
169        venv_dir = ensure_venv(venv_name, reqs, python_version=python_version)
170        python = str(venv_dir / "bin" / "python")
171        result = subprocess.run(
172            [python, "-c", test_script],
173            capture_output=True,
174            text=True,
175            timeout=60,
176        )
177        actual_passed = result.returncode == 0
178        error = result.stderr.strip() if not actual_passed else None
179    except Exception as exc:
180        actual_passed = False
181        error = str(exc)
182
183    return ProbeResult(
184        function_key=function_key,
185        python_version=python_version,
186        torch_version=torch_version,
187        expected_in_range=expected_in_range,
188        actual_passed=actual_passed,
189        error=error,
190    )

193def _extract_upper_bound(spec_str: str) -> Optional[str]:
194    """Extract the upper bound version from a PEP 440 specifier.
195
196    E.g., ">=3.11,<3.13" → "3.13", ">=2.8" → None (no upper bound).
197    """
198    for part in spec_str.split(","):
199        part = part.strip()
200        if part.startswith("<") and not part.startswith("<="):
201            return part[1:]
202        if part.startswith("<="):
203            return part[2:]
204    return None

207def _next_minor(version: str) -> str:
208    """Bump the minor version by 1. E.g., "3.12" → "3.13", "2.8" → "2.9"."""
209    parts = version.split(".")
210    parts[-1] = str(int(parts[-1]) + 1)
211    return ".".join(parts)

214def _versions_to_probe(entry: CompatibilityEntry) -> tuple[list[str], list[str]]:
215    """Determine which versions to probe for a function.
216
217    Strategy: test one version inside each boundary (should pass) and
218    one version just outside the upper bound (should fail). We don't
219    test below the lower bound — no need to support older versions.
220
221    Returns:
222        (python_versions, torch_versions) to probe.
223    """
224    py_versions = set()
225    torch_versions = set()
226
227    # Python: one inside (highest supported), one just above upper bound
228    py_upper = _extract_upper_bound(str(entry.python_versions))
229    if py_upper:
230        # Test the version just below the upper bound (should pass)
231        below = py_upper.split(".")
232        below[-1] = str(int(below[-1]) - 1)
233        py_versions.add(".".join(below))
234        # Test the upper bound itself (should fail)
235        py_versions.add(py_upper)
236    else:
237        # No upper bound — test latest known versions (should pass)
238        py_versions.update(["3.13", "3.14"])
239
240    # Torch: same logic
241    torch_upper = _extract_upper_bound(str(entry.torch_versions))
242    if torch_upper:
243        below = torch_upper.split(".")
244        below[-1] = str(int(below[-1]) - 1)
245        torch_versions.add(".".join(below))
246        torch_versions.add(torch_upper)
247    else:
248        torch_versions.update(["2.10"])  # latest known
249
250    return sorted(py_versions), sorted(torch_versions)

253def run_compatibility_probes(
254    functions: Optional[list[str]] = None,
255    python_versions: Optional[list[str]] = None,
256    torch_versions: Optional[list[str]] = None,
257) -> CompatibilityReport:
258    """Run compatibility probes at version boundaries.
259
260    For each function, probes ONE version inside the boundary (should pass)
261    and ONE version just outside the upper bound (should fail). Does NOT
262    test below the lower bound — we don't need to support older versions.
263
264    Override auto-detection with explicit python_versions/torch_versions
265    to probe specific combos.
266
267    Args:
268        functions: Function keys to test. Default: all non-isolated.
269        python_versions: Override: specific Python versions to test.
270        torch_versions: Override: specific torch versions to test.
271
272    Returns:
273        CompatibilityReport with results.
274    """
275    func_keys = functions or [k for k, v in COMPATIBILITY_MATRIX.items() if not v.isolated]
276    report = CompatibilityReport()
277
278    for func_key in func_keys:
279        entry = COMPATIBILITY_MATRIX[func_key]
280
281        if python_versions and torch_versions:
282            py_vers, t_vers = python_versions, torch_versions
283        else:
284            py_vers, t_vers = _versions_to_probe(entry)
285
286        for py_ver in py_vers:
287            for t_ver in t_vers:
288                logger.info("Probing %s on Python %s / torch %s", func_key, py_ver, t_ver)
289                result = _probe_function(func_key, py_ver, t_ver)
290                report.results.append(result)
291
292    return report

295def update_matrix_from_report(report: CompatibilityReport) -> dict[str, dict[str, str]]:
296    """Suggest matrix updates based on probe results.
297
298    Returns:
299        Dict mapping function_key → {"python_versions": new_spec, "torch_versions": new_spec}
300        Only includes entries that need updating.
301    """
302    updates: dict[str, dict[str, str]] = {}
303
304    # Group results by function
305    by_func: dict[str, list[ProbeResult]] = {}
306    for r in report.results:
307        by_func.setdefault(r.function_key, []).append(r)
308
309    for func_key, results in by_func.items():
310        passing_py = sorted({r.python_version for r in results if r.actual_passed})
311        failing_py = sorted({r.python_version for r in results if not r.actual_passed})
312        failing_torch = sorted({r.torch_version for r in results if not r.actual_passed})
313
314        entry = COMPATIBILITY_MATRIX.get(func_key)
315        if not entry:
316            continue
317
318        suggested: dict[str, str] = {}
319
320        # Suggest widened Python range if functions pass on higher versions
321        if passing_py:
322            max_passing = passing_py[-1]
323            current_spec = str(entry.python_versions)
324            if f"<{max_passing}" in current_spec or not entry.python_versions.contains(f"{max_passing}.0"):
325                suggested["python_versions"] = f">={passing_py[0]}"
326                if failing_py:
327                    min_failing = failing_py[0]
328                    suggested["python_versions"] += f",<{min_failing}"
329
330        # Suggest widened torch range
331        passing_t = sorted({r.torch_version for r in results if r.actual_passed})
332        if passing_t:
333            max_t = passing_t[-1]
334            current_t = str(entry.torch_versions)
335            if f"<{max_t}" in current_t or not entry.torch_versions.contains(f"{max_t}.0"):
336                suggested["torch_versions"] = f">={passing_t[0]}"
337                if failing_torch:
338                    suggested["torch_versions"] += f",<{failing_torch[0]}"
339
340        if suggested:
341            updates[func_key] = suggested
342
343    return updates

308        by_func.setdefault(r.function_key, []).append(r)
309
310    for func_key, results in by_func.items():
311        passing_py = sorted({r.python_version for r in results if r.actual_passed})
312        failing_py = sorted({r.python_version for r in results if not r.actual_passed})
313        failing_torch = sorted({r.torch_version for r in results if not r.actual_passed})
314
315        entry = COMPATIBILITY_MATRIX.get(func_key)
316        if not entry:
317            continue
318
319        suggested: dict[str, str] = {}
320
321        # Suggest widened Python range if functions pass on higher versions
322        if passing_py:
323            max_passing = passing_py[-1]
324            current_spec = str(entry.python_versions)
325            if f"<{max_passing}" in current_spec or not entry.python_versions.contains(f"{max_passing}.0"):
326                suggested["python_versions"] = f">={passing_py[0]}"
327                if failing_py:
328                    min_failing = failing_py[0]
329                    suggested["python_versions"] += f",<{min_failing}"
330
331        # Suggest widened torch range
332        passing_t = sorted({r.torch_version for r in results if r.actual_passed})
333        if passing_t:
334            max_t = passing_t[-1]
335            current_t = str(entry.torch_versions)
336            if f"<{max_t}" in current_t or not entry.torch_versions.contains(f"{max_t}.0"):
337                suggested["torch_versions"] = f">={passing_t[0]}"
338                if failing_torch:
339                    suggested["torch_versions"] += f",<{failing_torch[0]}"
340
341        if suggested:
342            updates[func_key] = suggested
