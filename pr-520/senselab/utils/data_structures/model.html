 27class SenselabModel(BaseModel, Generic[PROVIDER_T]):
 28    """Base configuration for SenselabModel class."""
 29
 30    path_or_uri: Union[str, Path]
 31    revision: Optional[str] = None
 32
 33    @field_validator("path_or_uri", mode="before")
 34    def validate_path_or_uri(cls, value: Union[str, Path]) -> Union[str, Path]:
 35        """Validate the path_or_uri.
 36
 37        This check is only for files and not for remote resources.
 38        It does check if the path_or_uri is not empty and if it is an
 39        existing file.
 40        """
 41        if not value:
 42            raise ValueError("path_or_uri cannot be empty")
 43
 44        if isinstance(value, Path) and not os.path.isfile(value):
 45            raise ValueError("path_or_uri is not an existing file")
 46
 47        # If the value is a string and looks like an existing file path,
 48        # convert it to a Path object
 49        if isinstance(value, str) and os.path.isfile(value):
 50            value = Path(value)
 51            if not is_torch_model(value):
 52                raise ValueError("path_or_uri does not point to a valid torch model")
 53
 54        return value

 57class HFModel(SenselabModel[PROVIDER_T]):
 58    """HuggingFace model.
 59
 60    Note: For some HuggingFace models, HF_TOKEN may be required for access.
 61    """
 62
 63    revision: Annotated[str, Field(validate_default=True)] = "main"
 64    info: Optional[ModelInfo] = None
 65    _hf_cache: ClassVar[Dict[Tuple[str, str], bool]] = {}
 66
 67    @field_validator("revision")
 68    def validate_hf_model_id(cls, value: str, info: ValidationInfo) -> Union[str, Path]:
 69        """Validate the path_or_uri.
 70
 71        This check is only for remote resources and not for files.
 72        It checks if the specified Hugging Face model ID and revision exist
 73        in the remote Hub.
 74        """
 75        path_or_uri = info.data["path_or_uri"]
 76        if not isinstance(path_or_uri, Path):
 77            if (str(path_or_uri), value) not in cls._hf_cache:
 78                cls._hf_cache[(str(path_or_uri), value)] = check_hf_repo_exists(
 79                    repo_id=str(path_or_uri), revision=value, repo_type="model"
 80                )
 81
 82            if not cls._hf_cache[(str(path_or_uri), value)]:
 83                raise ValueError(
 84                    f"The huggingface model: path_or_uri ({path_or_uri}) or "
 85                    f"specified revision ({value}) cannot be found.\n"
 86                    "Please check the model ID and revision. If the model is "
 87                    "private or restricted access, make sure you have access "
 88                    "to it and have exported your huggingface token in your "
 89                    "environment variables."
 90                )
 91        return value
 92
 93    def get_model_info(self) -> ModelInfo:
 94        """Gets the model info using the HuggingFace API and saves it as a property."""
 95        if isinstance(self.path_or_uri, Path):
 96            raise ValueError("Model info is only available for remote resources and not for files.")
 97        if not self.info:
 98            api = HfApi(token=get_huggingface_token())
 99            self.info = api.model_info(repo_id=self.path_or_uri, revision=self.revision)
100        return self.info

183def check_torchaudio_model_exists(model_id: str) -> bool:
184    """Private function to check if a torchaudio model exists."""
185    if not TORCHAUDIO_AVAILABLE:
186        raise ModuleNotFoundError(
187            "`torchaudio` is not installed. Please install senselab audio dependencies using `pip install senselab`."
188        )
189
190    try:
191        _ = getattr(torchaudio.pipelines, model_id)
192        return True
193    except AttributeError:
194        return False

198    """Check if a file is a torch model."""
199    try:
200        _ = torch.load(file_path)
201        return True
202    except Exception:
203        return False

206def get_huggingface_token(env_file_path: Optional[Union[str, Path]] = None) -> Optional[str]:
207    """Return a Hugging Face token from the environment or a local `.env` file."""
208    token_env_vars = ("HF_TOKEN", "HUGGING_FACE_HUB_TOKEN", "HUGGINGFACE_HUB_TOKEN")
209    for env_var in token_env_vars:
210        token = os.getenv(env_var)
211        if token:
212            return token
213
214    dotenv_path = Path(env_file_path).expanduser() if env_file_path is not None else Path(find_dotenv(usecwd=True))
215    if not dotenv_path.is_file():
216        return None
217
218    dotenv_values_dict = dotenv_values(dotenv_path)
219    for env_var in token_env_vars:
220        token = dotenv_values_dict.get(env_var)
221        if token:
222            return str(token)
223    return None

226def check_hf_repo_exists(repo_id: str, revision: str = "main", repo_type: str = "model") -> bool:
227    """Check if a Hugging Face repository exists.
228
229    For models, uses :func:`ensure_hf_model` which coordinates across
230    processes via file locking and caches results on the shared filesystem.
231    """
232    if repo_type == "model":
233        from senselab.utils.dependencies import ensure_hf_model
234
235        try:
236            ensure_hf_model(repo_id, revision)
237            return True
238        except Exception:
239            return False
240
241    # Non-model repos (rare): direct API check
242    api = HfApi(token=get_huggingface_token())
243    try:
244        api.list_repo_commits(repo_id=repo_id, revision=revision, repo_type=repo_type)
245        return True
246    except (RepositoryNotFoundError, RevisionNotFoundError):
247        return False

250@lru_cache(maxsize=128)
251def check_github_repo_exists(repo_id: str, branch: str = "main") -> bool:
252    """Checks if a GitHub repository exists with caching and authentication."""
253    url = f"https://api.github.com/repos/{repo_id}/branches/{branch}"
254    token = os.getenv("GITHUB_TOKEN") or None
255
256    headers = {}
257    if token:
258        headers = {"Authorization": f"token {token}"}
259
260    response = requests.get(url, headers=headers, timeout=10)
261
262    if response.status_code == 200:
263        return True
264    elif response.status_code == 404:
265        return False
266    elif response.status_code == 403:  # Handle rate limit exceeded
267        print("GitHub API rate limit exceeded. Please try again later.")
268        return False
269    else:
270        response.raise_for_status()
271        return False
