senselab.audio.tasks.voice_cloning.coqui

Voice cloning using Coqui TTS via isolated subprocess venv.

Coqui TTS has conflicting dependencies (pins old torch versions, requires Python <=3.11). It runs in an isolated subprocess venv managed by uv. Audio data is serialized as FLAC for efficient lossless transfer.

 15from senselab.audio.data_structures import Audio

 19# Coqui venv specification
 20_COQUI_VENV = "coqui"
 21_COQUI_REQUIREMENTS = [
 22    "coqui-tts~=0.27",
 23    "torch>=2.8,<2.9",
 24    "torchaudio>=2.8,<2.9",
 25    "transformers>=4.52,<5",  # coqui-tts 0.27 needs >=4.52; isin_mps_friendly removed in 5.0
 26    "numpy",
 27    "soundfile",
 28]
 29_COQUI_PYTHON = "3.11"
 30
 31# Worker script — runs inside the isolated venv (no senselab imports)
 32_WORKER_SCRIPT = r"""
 33import json
 34import sys
 35from pathlib import Path
 36
 37try:
 38    import soundfile as sf
 39    import torch
 40    from TTS.api import TTS
 41
 42    args = json.loads(sys.stdin.read())
 43    source_paths = args["source_paths"]
 44    target_paths = args["target_paths"]
 45    model_id = args["model_id"]
 46    device = args["device"]
 47    output_dir = args["output_dir"]
 48
 49    tts = TTS(model_id).to(device=device)
 50    audio_config = tts.voice_converter.vc_config.audio
 51
 52    expected_sample_rate = (
 53        audio_config.input_sample_rate
 54        if hasattr(audio_config, "input_sample_rate")
 55        else getattr(audio_config, "sample_rate", None)
 56    )
 57    output_sample_rate = (
 58        audio_config.output_sample_rate
 59        if hasattr(audio_config, "output_sample_rate")
 60        else getattr(audio_config, "sample_rate", expected_sample_rate)
 61    )
 62
 63    output_paths = []
 64    for i, (src_path, tgt_path) in enumerate(zip(source_paths, target_paths)):
 65        src_data, src_sr = sf.read(src_path, dtype="float32")
 66        tgt_data, tgt_sr = sf.read(tgt_path, dtype="float32")
 67        src_wav = torch.from_numpy(src_data).unsqueeze(0) if src_data.ndim == 1 else torch.from_numpy(src_data.T)
 68        tgt_wav = torch.from_numpy(tgt_data).unsqueeze(0) if tgt_data.ndim == 1 else torch.from_numpy(tgt_data.T)
 69
 70        if expected_sample_rate is not None:
 71            if src_sr != expected_sample_rate:
 72                raise ValueError(
 73                    f"[Pair {i}] Source sample rate {src_sr} != model expected {expected_sample_rate}"
 74                )
 75            if tgt_sr != expected_sample_rate:
 76                raise ValueError(
 77                    f"[Pair {i}] Target sample rate {tgt_sr} != model expected {expected_sample_rate}"
 78                )
 79
 80        converted = tts.voice_conversion(
 81            source_wav=src_wav.squeeze(),
 82            target_wav=tgt_wav.squeeze(),
 83        )
 84
 85        if not isinstance(converted, torch.Tensor):
 86            converted = torch.tensor(converted)
 87        if converted.dim() == 1:
 88            converted = converted.unsqueeze(0)
 89
 90        sr = output_sample_rate or src_sr
 91        out_path = str(Path(output_dir) / f"cloned_{i}.flac")
 92        sf.write(out_path, converted.squeeze().cpu().numpy(), sr, format="FLAC")
 93        output_paths.append(out_path)
 94
 95    print(json.dumps({"output_paths": output_paths}))
 96except Exception as exc:
 97    print(json.dumps({"error": {"type": type(exc).__name__, "message": str(exc)}}))
 98    sys.exit(1)
 99"""

102def list_coqui_models() -> list:
103    """List available Coqui TTS models via the isolated subprocess venv."""
104    venv_dir = ensure_venv(_COQUI_VENV, _COQUI_REQUIREMENTS, python_version=_COQUI_PYTHON)
105    python = str(venv_dir / "bin" / "python")
106    env = _clean_subprocess_env()
107    result = subprocess.run(
108        [python, "-c", "from TTS.api import TTS; import json; print(json.dumps(list(TTS().list_models())))"],
109        capture_output=True,
110        text=True,
111        timeout=300,  # cold start can be slow (venv creation + TTS model list download)
112        env=env,
113    )
114    if result.returncode != 0:
115        raise RuntimeError(f"Failed to list Coqui models:\n{result.stderr}")
116    return json.loads(result.stdout.strip().splitlines()[-1])

119class CoquiVoiceCloner:
120    """Voice cloning via Coqui TTS in an isolated subprocess venv."""
121
122    @classmethod
123    def clone_voices(
124        cls,
125        source_audios: List[Audio],
126        target_audios: List[Audio],
127        model: Optional[CoquiTTSModel] = None,
128        device: Optional[DeviceType] = None,
129    ) -> List[Audio]:
130        """Clone voices from source audios to target audios using Coqui TTS.
131
132        The actual Coqui TTS operations run in an isolated subprocess venv
133        with its own Python and torch version. Audio is transferred via
134        lossless FLAC files.
135
136        Args:
137            source_audios: List of source audio objects.
138            target_audios: List of target audio objects.
139            model: Coqui TTS model. Default: knnvc voice conversion.
140            device: Device preference (CUDA or CPU).
141        """
142        if model is None:
143            model = CoquiTTSModel(path_or_uri="voice_conversion_models/multilingual/multi-dataset/knnvc")
144
145        if len(source_audios) != len(target_audios):
146            raise ValueError("Number of source and target audios must be the same.")
147
148        venv_dir = ensure_venv(_COQUI_VENV, _COQUI_REQUIREMENTS, python_version=_COQUI_PYTHON)
149        python = str(venv_dir / "bin" / "python")
150
151        with tempfile.TemporaryDirectory(prefix="senselab-coqui-") as tmpdir:
152            tmp = Path(tmpdir)
153
154            # Validate and serialize in a single pass
155            source_paths = []
156            target_paths = []
157            for i, (src, tgt) in enumerate(zip(source_audios, target_audios)):
158                if src.waveform.squeeze().dim() != 1 or tgt.waveform.squeeze().dim() != 1:
159                    raise ValueError(f"[Pair {i}] Only mono audio is supported.")
160
161                src_path = str(tmp / f"source_{i}.flac")
162                tgt_path = str(tmp / f"target_{i}.flac")
163                src.save_to_file(src_path, format="flac")
164                tgt.save_to_file(tgt_path, format="flac")
165                source_paths.append(src_path)
166                target_paths.append(tgt_path)
167
168            # Run worker in isolated venv
169            input_json = json.dumps(
170                {
171                    "source_paths": source_paths,
172                    "target_paths": target_paths,
173                    "model_id": str(model.path_or_uri),
174                    "device": device.value if device else "cpu",
175                    "output_dir": str(tmp),
176                }
177            )
178
179            # Clear MPLBACKEND to avoid matplotlib_inline errors in subprocess
180            env = {k: v for k, v in os.environ.items() if k != "MPLBACKEND"}
181            result = subprocess.run(
182                [python, "-c", _WORKER_SCRIPT],
183                input=input_json,
184                capture_output=True,
185                text=True,
186                timeout=600,
187                env=env,
188            )
189
190            output = parse_subprocess_result(result, "Coqui")
191
192            # Load results eagerly (tempdir is cleaned up after this block)
193            cloned_audios = []
194            for out_path in output.get("output_paths", []):
195                audio = Audio(filepath=out_path)
196                _ = audio.waveform  # force load before tempdir cleanup
197                cloned_audios.append(audio)
198
199            return cloned_audios
