Features extraction

Task Overview

This module provides the API for extracting voice and speech features from audio recordings using the senselab package. Features span multiple speech subsystems/clincal constructs—such as fluency, respiration, phonation, articulation, and spectral characteristics—and are derived using trusted libraries including Praat-Parselmouth, OpenSMILE, Torchaudio, and Torchaudio-SQUIM.

270        return []

273    seconds_per_frame = total_duration / num_frames
274

289            segments.append(

292                    "phoneme": label,
293                    "start_frame": start_frame,
294                    "end_frame": frame - 1,

296                    "start_seconds": start_frame * seconds_per_frame,
297                    "end_seconds": frame * seconds_per_frame,
298                    "duration_seconds": frame_count * seconds_per_frame,
299                }
300            )
301            current_idx = idx
302            start_frame = frame
303
304    # Final segment

202def to_frame_major_posteriorgram(posteriorgram: torch.Tensor) -> torch.Tensor:
203    """Normalize a PPG tensor to frame-major layout ``(frames, phonemes)``.
204
205    The ppgs library typically returns tensors shaped ``(1, phonemes, frames)``
206    or ``(phonemes, frames)``.  This helper squeezes any leading batch
207    dimension and transposes so that rows correspond to time frames and
208    columns correspond to phonemes.
209
210    Args:
211        posteriorgram: PPG tensor in any of the common layouts.
212
213    Returns:
214        A 2-D tensor of shape ``(frames, phonemes)``.
215
216    Raises:
217        ValueError: If the tensor has fewer than 2 dimensions after squeezing.
218    """
219    t = posteriorgram.detach().cpu()
220    # Remove only the batch dimension (dim 0) if it's size 1
221    if t.ndim == 3 and t.shape[0] == 1:
222        t = t.squeeze(0)
223    if t.ndim < 2:
224        raise ValueError(f"Expected at least a 2-D posteriorgram after squeezing, got shape {t.shape}")
225    # The ppgs library outputs (phonemes, frames) where the phoneme count
226    # matches len(_PHONEME_LABELS).  Use that knowledge first; fall back to
227    # the "smaller dimension = phonemes" heuristic for unknown inventories.
228    n_phonemes = len(_PHONEME_LABELS)
229    if t.shape[0] == n_phonemes and t.shape[1] != n_phonemes:
230        # (phonemes, frames) -> (frames, phonemes)
231        t = t.T
232    elif t.shape[1] == n_phonemes and t.shape[0] != n_phonemes:
233        # Already (frames, phonemes)
234        pass
235    elif t.shape[0] < t.shape[1]:
236        # Fallback heuristic: smaller dim is phonemes
237        t = t.T
238    # else: ambiguous (e.g. square); assume already frame-major
239    return t
