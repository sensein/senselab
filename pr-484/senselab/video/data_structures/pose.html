 87class IndividualPose(BaseModel):
 88    """Data structure for the estimated pose of a single individual in an image.
 89
 90    Attributes:
 91        individual_index (int): Index of the individual in the detection result.
 92        normalized_landmarks (Dict[str, PoseLandmark]): Dictionary of body landmarks with normalized coordinates.
 93        world_landmarks (Optional[Dict[str, PoseLandmark]]): Dictionary of body landmarks with real-world coordinates.
 94    """
 95
 96    individual_index: int
 97    normalized_landmarks: Dict[str, PoseLandmark]
 98    world_landmarks: Optional[Dict[str, PoseLandmark]] = None
 99
100    def get_landmark_coordinates(self, landmark: str, world: bool = False) -> PoseLandmark:
101        """Retrieve coordinates for a specific landmark.
102
103        Args:
104            landmark (str): Name of the landmark (e.g., "right_ankle").
105            world (bool): Whether to retrieve world coordinates. Defaults to False.
106
107        Returns:
108            PoseLandmark: Object containing information on the specified landmark.
109
110        Raises:
111            ValueError: If the specified landmark is not found.
112        """
113        landmarks = self.world_landmarks if world else self.normalized_landmarks
114        if not landmarks:
115            raise ValueError("No landmarks available.")
116        if landmark not in landmarks:
117            raise ValueError(f"Landmark '{landmark}' not found. Available landmarks: {sorted(landmarks.keys())}")
118        return landmarks[landmark]

129
130    model_config = ConfigDict(arbitrary_types_allowed=True)
131
132    model: PoseModel
133    image: np.ndarray
134    individuals: List[IndividualPose]
135
136    @field_validator("image", mode="before")
137    def validate_image(cls, value: np.ndarray) -> np.ndarray:
138        """Ensures image is a 3D numpy array with three color channels."""
139        if not isinstance(value, np.ndarray):
140            raise TypeError("Field 'image' must be a NumPy array.")
141        if value.ndim != 3 or value.shape[2] != 3:
142            raise ValueError("Field 'image' must be a 3D array with three color channels (RGB).")
143        return value
144
145    def get_individual(self, individual_index: int) -> IndividualPose:
146        """Retrieve a specific individual's pose data.
147
148        Args:
149            individual_index (int): Index of the individual to retrieve.
150
151        Returns:
152            IndividualPose: Pose data for the specified individual.
153
154        Raises:
155            ValueError: If the index is invalid or no individuals are detected.
156        """
157        if individual_index >= len(self.individuals) or individual_index < 0:
158            raise ValueError(
159                f"Individual index {individual_index} is invalid. {len(self.individuals)} poses were estimated. "
160                f"Valid indices are {f'0 to {len(self.individuals) - 1}' if len(self.individuals) > 0 else 'none'}"
161            )
162        return self.individuals[individual_index]
