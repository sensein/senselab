senselab.audio.tasks.features_extraction.sparc

SPARC feature extraction via isolated subprocess venv.

speech-articulatory-coding (SPARC) has dependencies that may conflict with the main environment. It runs in an isolated subprocess venv managed by uv, shared with ppgs where possible.

 38# Worker script for feature extraction — runs inside the isolated venv
 39_FEATURE_WORKER_SCRIPT = r"""
 40import json
 41import sys
 42import traceback
 43from pathlib import Path
 44
 45import numpy as np
 46import soundfile as sf
 47import torch
 48from sparc import load_model
 49
 50args = json.loads(sys.stdin.read())
 51audio_paths = args["audio_paths"]
 52language = args["language"]
 53device = args["device"]
 54output_dir = args["output_dir"]
 55
 56coder = load_model(language, device=device)
 57
 58output_paths = []
 59for i, audio_path in enumerate(audio_paths):
 60    data, sr = sf.read(audio_path, dtype="float32")
 61    waveform = data.squeeze()
 62
 63    try:
 64        features = coder.encode(waveform)
 65        # Convert all tensors/arrays to numpy for serialization
 66        result = {}
 67        for key, val in features.items():
 68            if isinstance(val, torch.Tensor):
 69                result[key] = val.cpu().numpy().tolist()
 70            elif isinstance(val, np.ndarray):
 71                result[key] = val.tolist()
 72            elif isinstance(val, tuple):
 73                result[key] = [float(v) for v in val]
 74            elif isinstance(val, (np.integer, np.floating)):
 75                result[key] = val.item()
 76            else:
 77                result[key] = val
 78    except Exception as e:
 79        print(f"Error extracting SPARC features for audio {i}: {e}", file=sys.stderr)
 80        print(traceback.format_exc(), file=sys.stderr)
 81        result = {"error": str(e)}
 82
 83    out_path = str(Path(output_dir) / f"sparc_{i}.json")
 84    with open(out_path, "w") as f:
 85        json.dump(result, f)
 86    output_paths.append(out_path)
 87
 88print(json.dumps({"output_paths": output_paths, "expected_sr": coder.sr}))
 89"""

 92_DECODE_WORKER_SCRIPT = r"""
 93import json
 94import sys
 95import traceback
 96from pathlib import Path
 97
 98import numpy as np
 99import soundfile as sf
100import torch
101from sparc import load_model
102
103args = json.loads(sys.stdin.read())
104language = args["language"]
105device = args["device"]
106output_dir = args["output_dir"]
107feature_dir = args["feature_dir"]
108
109coder = load_model(language, device=device)
110
111try:
112    # Load features from numpy files
113    ema = np.load(str(Path(feature_dir) / "ema.npy"))
114    pitch = np.load(str(Path(feature_dir) / "pitch.npy"))
115    loudness = np.load(str(Path(feature_dir) / "loudness.npy"))
116    spk_emb = np.load(str(Path(feature_dir) / "spk_emb.npy"))
117
118    waveform = coder.decode(ema, pitch, loudness, spk_emb)
119
120    # Save output as FLAC
121    out_path = str(Path(output_dir) / "decoded.flac")
122    if isinstance(waveform, torch.Tensor):
123        wav_np = waveform.detach().cpu().numpy().squeeze()
124    else:
125        wav_np = np.asarray(waveform).squeeze()
126    sf.write(out_path, wav_np, coder.output_sr)
127
128    print(json.dumps({"output_path": out_path, "sample_rate": coder.output_sr}))
129except Exception as e:
130    print(f"Error decoding SPARC features: {e}", file=sys.stderr)
131    print(traceback.format_exc(), file=sys.stderr)
132    print(json.dumps({"error": {"type": type(e).__name__, "message": str(e)}}))
133    sys.exit(1)
134"""

176class SparcFeatureExtractor:
177    """A factory for managing feature extraction pipelines using SPARC."""

179    @classmethod
180    def extract_sparc_features(
181        cls,
182        audios: List[Audio],
183        lang: Optional[Language] = None,
184        device: Optional[DeviceType] = None,
185        resample: Optional[bool] = False,
186    ) -> List[Dict[str, torch.Tensor]]:
187        """Extract SPARC articulatory features from audios.
188
189        The SPARC model runs in an isolated subprocess venv with its own
190        Python and dependencies. Audio is transferred via FLAC files.
191
192        Args:
193            audios: List of audio objects.
194            lang: Language for the SPARC model. None means multi-language.
195            device: Device to use (CUDA or CPU).
196            resample: Whether to resample audios to the model's expected rate.
197
198        Returns:
199            List of feature dicts with keys: ema, loudness, pitch,
200            periodicity, pitch_stats, spk_emb, ft_len.
201        """
202        device, _ = _select_device_and_dtype(
203            user_preference=device, compatible_devices=[DeviceType.CUDA, DeviceType.CPU]
204        )
205
206        if lang is None:
207            used_language = "multi"
208        elif lang.name == "english":
209            used_language = "en+"
210        else:
211            raise ValueError(f"Language {lang.name} not supported. Supported: english or None (multi-language).")
212
213        venv_dir = ensure_venv(_SPARC_VENV, _SPARC_REQUIREMENTS, python_version=_SPARC_PYTHON)
214        python = str(venv_dir / "bin" / "python")
215
216        with tempfile.TemporaryDirectory(prefix="senselab-sparc-") as tmpdir:
217            tmp = Path(tmpdir)
218
219            # Serialize audios to FLAC
220            audio_paths = []
221            for i, audio in enumerate(audios):
222                if audio.waveform.squeeze().dim() != 1:
223                    raise ValueError(f"Only mono audio files are supported. Audio index: {i}")
224                path = str(tmp / f"audio_{i}.flac")
225                audio.save_to_file(path, format="flac")
226                audio_paths.append(path)
227
228            # Run worker in isolated venv
229            input_json = json.dumps(
230                {
231                    "audio_paths": audio_paths,
232                    "language": used_language,
233                    "device": device.value,
234                    "output_dir": str(tmp),
235                }
236            )
237
238            result = subprocess.run(
239                [python, "-c", _FEATURE_WORKER_SCRIPT],
240                input=input_json,
241                capture_output=True,
242                text=True,
243                timeout=600,
244            )
245
246            output = parse_subprocess_result(result, "SPARC")
247            expected_sr = output.get("expected_sr")
248
249            # Check sample rates (after we know what the model expects)
250            if not resample:
251                for i, audio in enumerate(audios):
252                    if expected_sr and audio.sampling_rate != expected_sr:
253                        raise ValueError(
254                            f"Expected sample rate {expected_sr}, got {audio.sampling_rate}. "
255                            f"Audio index: {i}. Set resample=True to auto-resample."
256                        )
257
258            # If resampling needed, redo with resampled audio
259            if resample and expected_sr:
260                needs_resample = any(a.sampling_rate != expected_sr for a in audios)
261                if needs_resample:
262                    resampled = resample_audios(audios, resample_rate=expected_sr)
263                    return cls.extract_sparc_features(resampled, lang=lang, device=device, resample=False)
264
265            # Load results
266            features_list = []
267            for out_path in output.get("output_paths", []):
268                with open(out_path) as f:
269                    raw = json.load(f)
270
271                if "error" in raw:
272                    features_list.append(
273                        {
274                            "ema": torch.tensor(torch.nan),
275                            "loudness": torch.tensor(torch.nan),
276                            "pitch": torch.tensor(torch.nan),
277                            "periodicity": torch.tensor(torch.nan),
278                            "pitch_stats": torch.tensor(torch.nan),
279                            "spk_emb": torch.tensor(torch.nan),
280                            "ft_len": torch.nan,
281                        }
282                    )
283                else:
284                    features = {}
285                    for key, val in raw.items():
286                        if isinstance(val, list):
287                            features[key] = torch.tensor(val)
288                        else:
289                            features[key] = val
290                    features_list.append(features)
291
292            return features_list

294    @classmethod
295    def decode_sparc_features(
296        cls,
297        features: Dict[str, torch.Tensor],
298        lang: Optional[Language] = None,
299        device: Optional[DeviceType] = None,
300    ) -> Audio:
301        """Resynthesize audio from SPARC articulatory features.
302
303        Takes the feature dict returned by ``extract_sparc_features`` and
304        runs ``coder.decode(ema, pitch, loudness, spk_emb)`` in the isolated
305        subprocess venv to produce a waveform.
306
307        Args:
308            features: Feature dict with keys ``ema``, ``pitch``, ``loudness``,
309                and ``spk_emb`` (torch.Tensor values, as returned by
310                ``extract_sparc_features``).
311            lang: Language for the SPARC model. None means multi-language.
312            device: Device to use (CUDA or CPU).
313
314        Returns:
315            Resynthesized Audio object.
316        """
317        device, _ = _select_device_and_dtype(
318            user_preference=device, compatible_devices=[DeviceType.CUDA, DeviceType.CPU]
319        )
320
321        if lang is None:
322            used_language = "multi"
323        elif lang.name == "english":
324            used_language = "en+"
325        else:
326            raise ValueError(f"Language {lang.name} not supported. Supported: english or None (multi-language).")
327
328        required_keys = {"ema", "pitch", "loudness", "spk_emb"}
329        missing = required_keys - set(features.keys())
330        if missing:
331            raise ValueError(f"Missing required feature keys: {missing}")
332
333        venv_dir = ensure_venv(_SPARC_VENV, _SPARC_REQUIREMENTS, python_version=_SPARC_PYTHON)
334        python = str(venv_dir / "bin" / "python")
335
336        with tempfile.TemporaryDirectory(prefix="senselab-sparc-decode-") as tmpdir:
337            tmp = Path(tmpdir)
338
339            # Serialize feature tensors as numpy files
340            feature_dir = tmp / "features"
341            feature_dir.mkdir()
342            for key in required_keys:
343                tensor = features[key]
344                if isinstance(tensor, torch.Tensor):
345                    np.save(str(feature_dir / f"{key}.npy"), tensor.cpu().numpy())
346                else:
347                    np.save(str(feature_dir / f"{key}.npy"), np.asarray(tensor))
348
349            input_json = json.dumps(
350                {
351                    "language": used_language,
352                    "device": device.value,
353                    "output_dir": str(tmp),
354                    "feature_dir": str(feature_dir),
355                }
356            )
357
358            result = subprocess.run(
359                [python, "-c", _DECODE_WORKER_SCRIPT],
360                input=input_json,
361                capture_output=True,
362                text=True,
363                timeout=600,
364                env=_clean_subprocess_env(),
365            )
366
367            output = parse_subprocess_result(result, "SPARC decode")
368
369            audio_path = output["output_path"]
370            sample_rate = output["sample_rate"]
371            logger.info("SPARC decode produced audio at %d Hz: %s", sample_rate, audio_path)
372
373            tmp_audio = Audio(filepath=audio_path)
374            # Force lazy-load while the temp file still exists, then
375            # create a clean Audio without a filepath pointing at the
376            # (about-to-be-deleted) temp directory.
377            return Audio(waveform=tmp_audio.waveform, sampling_rate=tmp_audio.sampling_rate)

409        else:
410            raise ValueError(f"Language {lang.name} not supported. Supported: english or None (multi-language).")
411
412        for label, audio in [("source", source_audio), ("target", target_audio)]:
413            if audio.waveform.squeeze().dim() != 1:
414                raise ValueError(f"Only mono audio is supported ({label}).")
415
416        venv_dir = ensure_venv(_SPARC_VENV, _SPARC_REQUIREMENTS, python_version=_SPARC_PYTHON)

420            tmp = Path(tmpdir)
421
422            src_path = str(tmp / "source.wav")
423            trg_path = str(tmp / "target.wav")
424            source_audio.save_to_file(src_path, format="wav")
425            target_audio.save_to_file(trg_path, format="wav")
426
427            input_json = json.dumps(
428                {

434                }
435            )
436

438                [python, "-c", _CONVERT_WORKER_SCRIPT],
439                input=input_json,
440                capture_output=True,
441                text=True,
442                timeout=600,
443                env=_clean_subprocess_env(),
444            )
445
446            output = parse_subprocess_result(result, "SPARC convert")
447
448            audio_path = output["output_path"]
449            sample_rate = output["sample_rate"]

188        """Extract SPARC articulatory features from audios.
189
190        The SPARC model runs in an isolated subprocess venv with its own
191        Python and dependencies. Audio is transferred via FLAC files.
192
193        Args:
194            audios: List of audio objects.
195            lang: Language for the SPARC model. None means multi-language.
196            device: Device to use (CUDA or CPU).
197            resample: Whether to resample audios to the model's expected rate.
198
199        Returns:
200            List of feature dicts with keys: ema, loudness, pitch,
201            periodicity, pitch_stats, spk_emb, ft_len.
202        """
203        device, _ = _select_device_and_dtype(
204            user_preference=device, compatible_devices=[DeviceType.CUDA, DeviceType.CPU]
205        )
206
207        if lang is None:
208            used_language = "multi"
209        elif lang.name == "english":
210            used_language = "en+"
211        else:
212            raise ValueError(f"Language {lang.name} not supported. Supported: english or None (multi-language).")
213
214        venv_dir = ensure_venv(_SPARC_VENV, _SPARC_REQUIREMENTS, python_version=_SPARC_PYTHON)
215        python = str(venv_dir / "bin" / "python")
216
217        with tempfile.TemporaryDirectory(prefix="senselab-sparc-") as tmpdir:
218            tmp = Path(tmpdir)
219
220            # Serialize audios to FLAC
221            audio_paths = []
222            for i, audio in enumerate(audios):
223                if audio.waveform.squeeze().dim() != 1:
224                    raise ValueError(f"Only mono audio files are supported. Audio index: {i}")
225                path = str(tmp / f"audio_{i}.flac")
226                audio.save_to_file(path, format="flac")
227                audio_paths.append(path)
228
229            # Run worker in isolated venv
230            input_json = json.dumps(
231                {
232                    "audio_paths": audio_paths,
233                    "language": used_language,
234                    "device": device.value,
235                    "output_dir": str(tmp),
236                }
237            )
238
239            result = subprocess.run(
240                [python, "-c", _FEATURE_WORKER_SCRIPT],
241                input=input_json,
242                capture_output=True,
243                text=True,
244                timeout=600,
245            )
246
247            output = parse_subprocess_result(result, "SPARC")
248            expected_sr = output.get("expected_sr")
249
250            # Check sample rates (after we know what the model expects)
251            if not resample:
252                for i, audio in enumerate(audios):
253                    if expected_sr and audio.sampling_rate != expected_sr:
254                        raise ValueError(
255                            f"Expected sample rate {expected_sr}, got {audio.sampling_rate}. "
256                            f"Audio index: {i}. Set resample=True to auto-resample."
257                        )
258
259            # If resampling needed, redo with resampled audio
260            if resample and expected_sr:
261                needs_resample = any(a.sampling_rate != expected_sr for a in audios)
262                if needs_resample:
263                    resampled = resample_audios(audios, resample_rate=expected_sr)
264                    return cls.extract_sparc_features(resampled, lang=lang, device=device, resample=False)
265
266            # Load results
267            features_list = []
268            for out_path in output.get("output_paths", []):
269                with open(out_path) as f:
270                    raw = json.load(f)
271
272                if "error" in raw:
273                    features_list.append(
274                        {
275                            "ema": torch.tensor(torch.nan),
276                            "loudness": torch.tensor(torch.nan),
277                            "pitch": torch.tensor(torch.nan),
278                            "periodicity": torch.tensor(torch.nan),
279                            "pitch_stats": torch.tensor(torch.nan),
280                            "spk_emb": torch.tensor(torch.nan),
281                            "ft_len": torch.nan,
282                        }
283                    )
284                else:
285                    features = {}
286                    for key, val in raw.items():
287                        if isinstance(val, list):
288                            features[key] = torch.tensor(val)
289                        else:
290                            features[key] = val
291                    features_list.append(features)
292
293            return features_list
