senselab.audio.tasks.preprocessing.preprocessing

Audio preprocessing utilities implemented with Pydra workflows.

This module provides parallelizable preprocessing primitives operating on in-memory senselab.audio.data_structures.Audio objects. Each function exposes a simple API and runs as a small Pydra workflow so that batches can be processed concurrently (using the selected plugin).

Notes:
  • These functions operate on Audio objects (already loaded in memory); they do not read or write files.
  • For execution backends, see Pydra v1 plugins: https://nipype.github.io/pydra/
Common plugins:
  • "serial" / "debug" — sequential execution (default).
  • "cf" — concurrent futures for parallel execution.
  • "slurm" — submit tasks to a SLURM cluster.

Use plugin_args to tune the backend, e.g. {"n_procs": 8} for "cf".

 47# Resample Audios
 48# ---------------------------------------------------------------------
 49def resample_audios(
 50    audios: List[Audio],
 51    resample_rate: int,
 52    lowcut: Optional[float] = None,
 53    order: int = 4,
 54    cache_dir: Optional[Union[str, os.PathLike]] = None,
 55    plugin: str = "debug",
 56    plugin_args: Optional[Dict[str, Any]] = None,
 57) -> List[Audio]:
 58    """Resample a batch of `Audio` objects to a target sampling rate via Pydra.
 59
 60    For each channel, a zero-phase IIR low-pass filter (Butterworth, SOS) is applied,
 61    then resampling is performed with `speechbrain.augment.time_domain.Resample`.
 62
 63    Args:
 64        audios (list[Audio]):
 65            Input audio objects to be resampled.
 66        resample_rate (int):
 67            Target sampling rate in Hz.
 68        lowcut (float, optional):
 69            Low-pass cutoff frequency in Hz (applied before resampling). If ``None``,
 70            defaults to ``resample_rate / 2 - 100`` to avoid Nyquist artifacts.
 71        order (int, optional):
 72            Butterworth filter order (default: 4).
 73        cache_dir (str | os.PathLike, optional):
 74            Pydra cache directory. If ``None``, Pydra uses its default.
 75        plugin (str, optional):
 76            Pydra execution plugin. Common options:
 77              * ``"serial"`` / ``"debug"``: sequential (default).
 78              * ``"cf"``: concurrent futures (parallel).
 79              * ``"slurm"``: SLURM scheduler.
 80        plugin_args (dict, optional):
 81            Extra options for the chosen plugin (e.g., ``{"n_procs": 8}`` for "cf").
 82
 83    Returns:
 84        list[Audio]: Resampled `Audio` objects (sampling rate set to `resample_rate`).
 85
 86    Raises:
 87        ModuleNotFoundError:
 88            If `speechbrain` is not installed.
 89        ValueError:
 90            If provided parameters are invalid.
 91
 92    Example:
 93        >>> from senselab.audio.tasks.preprocessing import resample_audios
 94        >>> from senselab.audio.data_structures import Audio
 95        >>> a1 = Audio(filepath=Path("sample1.wav").resolve())
 96        >>> out = resample_audios([a1], resample_rate=16000, plugin="cf")
 97        >>> out[0].sampling_rate
 98        16000
 99    """
100    if not SPEECHBRAIN_AVAILABLE:
101        raise ModuleNotFoundError(
102            "`speechbrain` is not installed. "
103            "Please install senselab audio dependencies using `pip install senselab`."
104        )
105    plugin_args = plugin_args or {}
106
107    @python.define
108    def _resample(audio: Audio, resample_rate: int, lowcut: Optional[float], order: int) -> Audio:
109        # Defensive copy of metadata
110        md = audio.metadata.copy()
111
112        # Design low-pass if not provided
113        if lowcut is None:
114            lowcut = resample_rate / 2 - 100.0
115        sos = signal.butter(order, lowcut, btype="low", output="sos", fs=resample_rate)
116
117        # Per-channel IIR (zero-phase) then resample
118        resampler = Resample(orig_freq=audio.sampling_rate, new_freq=resample_rate)
119        channels = []
120        for ch in audio.waveform:
121            # sosfiltfilt works on numpy
122            filtered = signal.sosfiltfilt(sos, ch.detach().cpu().numpy()).copy()
123            filtered_t = torch.from_numpy(filtered).float().unsqueeze(0)  # [1, T]
124            res_ch = resampler(filtered_t).squeeze(0)  # [T']
125            channels.append(res_ch)
126
127        resampled_waveform = torch.stack(channels, dim=0)  # [C, T']
128        return Audio(waveform=resampled_waveform, sampling_rate=resample_rate, metadata=md)
129
130    @workflow.define
131    def _wf(xs: Sequence[Audio], resample_rate: int, lowcut: Optional[float], order: int) -> List[Audio]:
132        node = workflow.add(
133            _resample(resample_rate=resample_rate, lowcut=lowcut, order=order).split(audio=xs),
134            name="map_resample_audio",
135        )
136        return node.out
137
138    worker = _select_worker(plugin)
139    res = _wf(xs=audios, resample_rate=resample_rate, lowcut=lowcut, order=order)(
140        worker=worker, cache_root=cache_dir, **plugin_args
141    )
142    return list(res.out)

146# Downmix to Mono
147# ---------------------------------------------------------------------
148def downmix_audios_to_mono(
149    audios: List[Audio],
150    cache_dir: Optional[Union[str, os.PathLike]] = None,
151    plugin: str = "debug",
152    plugin_args: Optional[Dict[str, Any]] = None,
153) -> List[Audio]:
154    """Downmix each `Audio` to mono by averaging channels (parallel via Pydra).
155
156    Channel averaging preserves the original sampling rate and copies metadata.
157
158    Args:
159        audios (list[Audio]):
160            Input `Audio` objects. Works for mono and multi-channel inputs.
161        cache_dir (str | os.PathLike, optional):
162            Pydra cache directory. If ``None``, Pydra uses its default.
163        plugin (str, optional):
164            Pydra execution plugin. Common options:
165              * ``"serial"`` / ``"debug"``: sequential (default).
166              * ``"cf"``: concurrent futures (parallel).
167              * ``"slurm"``: SLURM scheduler.
168        plugin_args (dict, optional):
169            Extra options for the chosen plugin (e.g., ``{"n_procs": 8}``).
170
171    Returns:
172        list[Audio]: Mono `Audio` objects (shape [1, T]).
173
174    Example:
175        >>> from senselab.audio.tasks.preprocessing import downmix_audios_to_mono
176        >>> from senselab.audio.data_structures import Audio
177        >>> a1 = Audio(filepath=Path("sample1.wav").resolve())
178        >>> out = downmix_audios_to_mono([a1], plugin="cf")
179        >>> out[0].waveform.shape[0]
180        1
181    """
182    plugin_args = plugin_args or {}
183
184    @python.define
185    def _downmix(audio: Audio) -> Audio:
186        return Audio(
187            waveform=audio.waveform.mean(dim=0, keepdim=True),
188            sampling_rate=audio.sampling_rate,
189            metadata=audio.metadata.copy(),
190        )
191
192    @workflow.define
193    def _wf(xs: Sequence[Audio]) -> List[Audio]:
194        node = workflow.add(_downmix().split(audio=xs), name="map_downmix_audio")
195        return node.out
196
197    worker = _select_worker(plugin)
198    res = _wf(xs=audios)(worker=worker, cache_root=cache_dir, **plugin_args)
199    return list(res.out)

214    Args:

217        channel_index (int):
218            Zero-based channel index to extract.
219        cache_dir (str | os.PathLike, optional):
220            Pydra cache directory.
221        plugin (str, optional):
222            Pydra execution plugin. Common options:
223              * ``"serial"`` / ``"debug"``: sequential (default).
224              * ``"cf"``: concurrent futures (parallel).
225              * ``"slurm"``: SLURM scheduler.
226        plugin_args (dict, optional):
227            Extra options for the chosen plugin.
228
229    Returns:
230        list[Audio]: `Audio` objects containing only the selected channel (shape [1, T]).
231
232    Raises:
233        ValueError:
234            If `channel_index` is out of range for any input.

238        >>> from senselab.audio.data_structures import Audio

245
246    @python.define

248        if channel_index < 0 or audio.waveform.size(0) <= channel_index:

252
253    @workflow.define
254    def _wf(xs: Sequence[Audio], channel_index: int) -> List[Audio]:
255        node = workflow.add(_select(channel_index=channel_index).split(audio=xs), name="map_select_channel")
256        return node.out
257
258    worker = _select_worker(plugin)
259    res = _wf(xs=audios, channel_index=channel_index)(worker=worker, cache_root=cache_dir, **plugin_args)
260    return list(res.out)

263# ---------------------------------------------------------------------
264# Chunk Audios (by single [start, end]) via Pydra
265# ---------------------------------------------------------------------
266def chunk_audios(
267    data: Sequence[Tuple[Audio, Tuple[float, float]]],
268    cache_dir: Optional[Union[str, "os.PathLike"]] = None,
269    plugin: str = "debug",
270    plugin_args: Optional[Dict[str, Any]] = None,
271) -> List[Audio]:
272    """Extract a single time chunk from each `Audio` via a Pydra workflow.
273
274    Each pair provides the `(start, end)` times (in seconds) of the desired chunk.
275
276    Args:
277        data (Sequence[tuple[Audio, tuple[float, float]]]):
278            Sequence like ``[(audio, (start, end)), ...]``.
279        cache_dir (str | os.PathLike, optional):
280            Pydra cache directory.
281        plugin (str, optional):
282            Pydra execution plugin. Common options:
283              * ``"serial"`` / ``"debug"``: sequential (default).
284              * ``"cf"``: concurrent futures (parallel).
285              * ``"slurm"``: SLURM scheduler.
286        plugin_args (dict, optional):
287            Extra options for the chosen plugin.
288
289    Returns:
290        list[Audio]: One chunked `Audio` per input pair.
291
292    Raises:
293        ValueError:
294            If `start < 0`, `start >= end`, or `end` exceeds audio duration.
295
296    Example:
297        >>> from senselab.audio.tasks.preprocessing import chunk_audios
298        >>> from senselab.audio.data_structures import Audio
299        >>> a1 = Audio(filepath=Path("sample1.wav").resolve())
300        >>> pairs = [(a1, (0.5, 2.0)), (a1, (1.0, 3.0))]
301        >>> chunks = chunk_audios(pairs, plugin="cf", plugin_args={"n_procs": 8})
302        >>> len(chunks)
303        2
304    """
305    plugin_args = plugin_args or {}
306
307    @python.define
308    def _chunk_pair(pair: Tuple[Audio, Tuple[float, float]]) -> Audio:
309        """Chunk a single (Audio, (start, end)) pair."""
310        audio, (start, end) = pair
311
312        if start < 0:
313            raise ValueError("Start time must be >= 0.")
314        if start >= end:
315            raise ValueError(f"Start time ({start}) must be < end time ({end}).")
316
317        sr = audio.sampling_rate
318        wf = audio.waveform  # shape: (C, T)
319        duration = wf.shape[1] / sr
320        if end > duration:
321            raise ValueError(f"End time ({end}) must be <= audio duration ({duration:.6f} s).")
322
323        i0 = int(start * sr)
324        i1 = int(end * sr)
325        chunked_wf = wf[:, i0:i1]
326
327        return Audio(
328            waveform=chunked_wf,
329            sampling_rate=sr,
330            metadata={**audio.metadata, "segment_start": float(start), "segment_end": float(end)},
331        )
332
333    @workflow.define
334    def _wf(pairs: Sequence[Tuple[Audio, Tuple[float, float]]]) -> List[Audio]:
335        node = workflow.add(
336            _chunk_pair().split(pair=pairs),
337            name="map_chunk_audio",
338        )
339        return node.out  # list[Audio]
340
341    worker = "debug" if plugin in ("serial", "debug") else plugin
342    res = _wf(pairs=data)(worker=worker, cache_root=cache_dir, **plugin_args)
343    return list(res.out)

347# Extract multiple segments per audio
348# ---------------------------------------------------------------------
349def extract_segments(
350    data: List[Tuple[Audio, List[Tuple[float, float]]]],
351    cache_dir: Optional[Union[str, os.PathLike]] = None,
352    plugin: str = "debug",
353    plugin_args: Optional[Dict[str, Any]] = None,
354) -> List[List[Audio]]:
355    """Extract multiple time segments per `Audio` (maps over inputs via Pydra).
356
357    For each input `Audio`, a list of `(start, end)` segments (in seconds) is provided,
358    and the function returns a list of chunked `Audio` objects.
359
360    Args:
361        data (list[tuple[Audio, list[tuple[float, float]]]]):
362            Items like ``(audio, [(s1, e1), (s2, e2), ...])``.
363        cache_dir (str | os.PathLike, optional):
364            Pydra cache directory.
365        plugin (str, optional):
366            Pydra execution plugin. Common options:
367              * ``"serial"`` / ``"debug"``: sequential (default).
368              * ``"cf"``: concurrent futures (parallel).
369              * ``"slurm"``: SLURM scheduler.
370        plugin_args (dict, optional):
371            Extra options for the chosen plugin.
372
373    Returns:
374        list[list[Audio]]: For each input audio, a list of extracted segments.
375
376    Raises:
377        ValueError:
378            If any segment is invalid or exceeds audio duration.
379
380    Example:
381        >>> from senselab.audio.data_structures import Audio
382        >>> from senselab.audio.tasks.preprocessing import extract_segments
383        >>> a1 = Audio(filepath=Path("sample1.wav").resolve())
384        >>> specs = [(a1, [(0.0, 1.0), (1.5, 2.0)])]
385        >>> segmented = extract_segments(specs, plugin="cf")
386        >>> len(segmented[0])
387        2
388    """
389    plugin_args = plugin_args or {}
390
391    @python.define
392    def _extract(audio: Audio, timestamps: List[Tuple[float, float]]) -> List[Audio]:
393        sr = audio.sampling_rate
394        dur = audio.waveform.shape[1] / sr
395        out: List[Audio] = []
396        for start, end in timestamps:
397            if start < 0:
398                raise ValueError("Start time must be >= 0.")
399            if end > dur:
400                raise ValueError(f"End must be <= duration of the audio ({dur} sec).")
401            s = int(start * sr)
402            e = int(end * sr)
403            wf = audio.waveform[:, s:e]
404            out.append(Audio(waveform=wf, sampling_rate=sr, metadata=audio.metadata.copy()))
405        return out
406
407    @workflow.define
408    def _wf(pairs: Sequence[Tuple[Audio, List[Tuple[float, float]]]]) -> List[List[Audio]]:
409        # We split separately for the two items in the tuple by providing two sequences
410        audios = [p[0] for p in pairs]
411        tss = [p[1] for p in pairs]
412        node = workflow.add(_extract().split(audio=audios, timestamps=tss), name="map_extract_segments")
413        return node.out  # -> List[List[Audio]]
414
415    worker = _select_worker(plugin)
416    res = _wf(pairs=data)(worker=worker, cache_root=cache_dir, **plugin_args)
417    return list(res.out)

421# Pad Audios to desired length (samples)
422# ---------------------------------------------------------------------
423def pad_audios(
424    audios: List[Audio],
425    desired_samples: int,
426    cache_dir: Optional[Union[str, os.PathLike]] = None,
427    plugin: str = "debug",
428    plugin_args: Optional[Dict[str, Any]] = None,
429) -> List[Audio]:
430    """Right-pad each audio to a desired number of samples (parallel via Pydra).
431
432    Pads with zeros to reach `desired_samples`. If an audio is already longer or equal,
433    it is returned unchanged.
434
435    Args:
436        audios (list[Audio]):
437            Input `Audio` objects.
438        desired_samples (int):
439            Target length in samples (per channel).
440        cache_dir (str | os.PathLike, optional):
441            Pydra cache directory.
442        plugin (str, optional):
443            Pydra execution plugin. Common options:
444              * ``"serial"`` / ``"debug"``: sequential (default).
445              * ``"cf"``: concurrent futures (parallel).
446              * ``"slurm"``: SLURM scheduler.
447        plugin_args (dict, optional):
448            Extra options for the chosen plugin.
449
450    Returns:
451        list[Audio]: Padded `Audio` objects.
452
453    Example:
454        >>> from senselab.audio.data_structures import Audio
455        >>> from senselab.audio.tasks.preprocessing import pad_audios
456        >>> a1 = Audio(filepath=Path("sample1.wav").resolve())
457        >>> padded = pad_audios([a1], desired_samples=16000, plugin="cf")
458        >>> padded[0].waveform.shape[1] >= 16000
459        True
460    """
461    plugin_args = plugin_args or {}
462
463    @python.define
464    def _pad(audio: Audio, desired_samples: int) -> Audio:
465        current = audio.waveform.shape[1]
466        if current >= desired_samples:
467            return audio
468        padding_needed = desired_samples - current
469        padded_waveform = torch.nn.functional.pad(audio.waveform, (0, padding_needed))
470        return Audio(
471            waveform=padded_waveform,
472            sampling_rate=audio.sampling_rate,
473            metadata=audio.metadata.copy(),
474        )
475
476    @workflow.define
477    def _wf(xs: Sequence[Audio], desired_samples: int) -> List[Audio]:
478        node = workflow.add(_pad(desired_samples=desired_samples).split(audio=xs), name="map_pad_audio")
479        return node.out
480
481    worker = _select_worker(plugin)
482    res = _wf(xs=audios, desired_samples=desired_samples)(worker=worker, cache_root=cache_dir, **plugin_args)
483    return list(res.out)

487# Evenly segment audios (optionally pad last)
488# ---------------------------------------------------------------------
489def evenly_segment_audios(
490    audios: List[Audio],
491    segment_length: float,
492    pad_last_segment: bool = True,
493    cache_dir: Optional[Union[str, os.PathLike]] = None,
494    plugin: str = "debug",
495    plugin_args: Optional[Dict[str, Any]] = None,
496) -> List[List[Audio]]:
497    """Split each audio into fixed-length segments (seconds), optionally padding the last.
498
499    If `pad_last_segment` is True and the final segment is shorter than `segment_length`,
500    it is right-padded with zeros to match the target duration.
501
502    Args:
503        audios (list[Audio]):
504            Input `Audio` objects.
505        segment_length (float):
506            Segment duration in seconds.
507        pad_last_segment (bool, optional):
508            Whether to pad the last segment to full length (default: True).
509        cache_dir (str | os.PathLike, optional):
510            Pydra cache directory.
511        plugin (str, optional):
512            Pydra execution plugin. Common options:
513              * ``"serial"`` / ``"debug"``: sequential (default).
514              * ``"cf"``: concurrent futures (parallel).
515              * ``"slurm"``: SLURM scheduler.
516        plugin_args (dict, optional):
517            Extra options for the chosen plugin.
518
519    Returns:
520        list[list[Audio]]: For each input, a list of fixed-length segments.
521
522    Example:
523        >>> from senselab.audio.data_structures import Audio
524        >>> from senselab.audio.tasks.preprocessing import evenly_segment_audios
525        >>> a1 = Audio(filepath=Path("sample1.wav").resolve())
526        >>> segs = evenly_segment_audios([a1], segment_length=1.0, pad_last_segment=True, plugin="cf")
527        >>> len(segs[0]) >= 1
528        True
529    """
530    plugin_args = plugin_args or {}
531
532    @python.define
533    def _even_segments(audio: Audio, segment_length: float, pad_last_segment: bool) -> List[Audio]:
534        sr = audio.sampling_rate
535        total_samples = audio.waveform.shape[1]
536        seg_samples = max(1, int(round(segment_length * sr)))
537        segments: List[Audio] = []
538
539        if total_samples == 0:
540            return [Audio(waveform=audio.waveform.clone(), sampling_rate=sr, metadata=audio.metadata.copy())]
541
542        # Full segments
543        n_full = total_samples // seg_samples
544        for i in range(n_full):
545            s = i * seg_samples
546            e = s + seg_samples
547            wf = audio.waveform[:, s:e]
548            segments.append(Audio(waveform=wf, sampling_rate=sr, metadata=audio.metadata.copy()))
549
550        # Remainder
551        rem = total_samples % seg_samples
552        if rem > 0:
553            s = n_full * seg_samples
554            wf = audio.waveform[:, s:total_samples]
555            if pad_last_segment:
556                pad_needed = seg_samples - wf.shape[1]
557                wf = torch.nn.functional.pad(wf, (0, pad_needed))
558            segments.append(Audio(waveform=wf, sampling_rate=sr, metadata=audio.metadata.copy()))
559
560        # Edge case: if audio shorter than one segment and no remainder branch executed (because rem==total)
561        if len(segments) == 0:
562            wf = audio.waveform
563            if pad_last_segment and wf.shape[1] < seg_samples:
564                pad_needed = seg_samples - wf.shape[1]
565                wf = torch.nn.functional.pad(wf, (0, pad_needed))
566            segments.append(Audio(waveform=wf, sampling_rate=sr, metadata=audio.metadata.copy()))
567
568        return segments
569
570    @workflow.define
571    def _wf(xs: Sequence[Audio], segment_length: float, pad_last_segment: bool) -> List[List[Audio]]:
572        node = workflow.add(
573            _even_segments(segment_length=segment_length, pad_last_segment=pad_last_segment).split(audio=xs),
574            name="map_evenly_segment",
575        )
576        return node.out  # -> List[List[Audio]]
577
578    worker = _select_worker(plugin)
579    res = _wf(xs=audios, segment_length=segment_length, pad_last_segment=pad_last_segment)(
580        worker=worker, cache_root=cache_dir, **plugin_args
581    )
582    return list(res.out)

588def concatenate_audios(audios: List[Audio]) -> Audio:
589    """Concatenate multiple `Audio` objects along the time axis.

112
113        # Design low-pass if not provided

498    """Split each audio into fixed-length segments (seconds), optionally padding the last.
499
500    If `pad_last_segment` is True and the final segment is shorter than `segment_length`,
501    it is right-padded with zeros to match the target duration.
502
503    Args:
504        audios (list[Audio]):
505            Input `Audio` objects.
506        segment_length (float):
507            Segment duration in seconds.
508        pad_last_segment (bool, optional):
509            Whether to pad the last segment to full length (default: True).
510        cache_dir (str | os.PathLike, optional):
511            Pydra cache directory.
512        plugin (str, optional):
513            Pydra execution plugin. Common options:
514              * ``"serial"`` / ``"debug"``: sequential (default).
515              * ``"cf"``: concurrent futures (parallel).
516              * ``"slurm"``: SLURM scheduler.
517        plugin_args (dict, optional):
518            Extra options for the chosen plugin.
519
520    Returns:
521        list[list[Audio]]: For each input, a list of fixed-length segments.
522
523    Example:
524        >>> from senselab.audio.data_structures import Audio
525        >>> from senselab.audio.tasks.preprocessing import evenly_segment_audios
526        >>> a1 = Audio(filepath=Path("sample1.wav").resolve())
527        >>> segs = evenly_segment_audios([a1], segment_length=1.0, pad_last_segment=True, plugin="cf")
528        >>> len(segs[0]) >= 1
529        True
530    """
531    plugin_args = plugin_args or {}
532
533    @python.define
534    def _even_segments(audio: Audio, segment_length: float, pad_last_segment: bool) -> List[Audio]:
535        sr = audio.sampling_rate
536        total_samples = audio.waveform.shape[1]
537        seg_samples = max(1, int(round(segment_length * sr)))
538        segments: List[Audio] = []
539
540        if total_samples == 0:
541            return [Audio(waveform=audio.waveform.clone(), sampling_rate=sr, metadata=audio.metadata.copy())]
542
543        # Full segments
544        n_full = total_samples // seg_samples
545        for i in range(n_full):
546            s = i * seg_samples
547            e = s + seg_samples
548            wf = audio.waveform[:, s:e]
549            segments.append(Audio(waveform=wf, sampling_rate=sr, metadata=audio.metadata.copy()))
550
551        # Remainder
552        rem = total_samples % seg_samples
553        if rem > 0:
554            s = n_full * seg_samples
555            wf = audio.waveform[:, s:total_samples]
556            if pad_last_segment:
557                pad_needed = seg_samples - wf.shape[1]
558                wf = torch.nn.functional.pad(wf, (0, pad_needed))
559            segments.append(Audio(waveform=wf, sampling_rate=sr, metadata=audio.metadata.copy()))
560
561        # Edge case: if audio shorter than one segment and no remainder branch executed (because rem==total)
562        if len(segments) == 0:
563            wf = audio.waveform
564            if pad_last_segment and wf.shape[1] < seg_samples:
565                pad_needed = seg_samples - wf.shape[1]
566                wf = torch.nn.functional.pad(wf, (0, pad_needed))
567            segments.append(Audio(waveform=wf, sampling_rate=sr, metadata=audio.metadata.copy()))
568
569        return segments
570
571    @workflow.define
572    def _wf(xs: Sequence[Audio], segment_length: float, pad_last_segment: bool) -> List[List[Audio]]:
573        node = workflow.add(
574            _even_segments(segment_length=segment_length, pad_last_segment=pad_last_segment).split(audio=xs),
575            name="map_evenly_segment",
576        )
577        return node.out  # -> List[List[Audio]]
578

580    res = _wf(xs=audios, segment_length=segment_length, pad_last_segment=pad_last_segment)(
581        worker=worker, cache_root=cache_dir, **plugin_args
582    )
583    return list(res.out)
