 17class Participant(BaseModel):

 34        """Overloads the default BaseModel equality to correctly check that ids are equivalent."""

 86    def generate_audios_from_filepaths(cls, v: Union[List[str], List[Audio]], _: ValidationInfo) -> List[Audio]:
 87        """Generate the audios in the dataset from a list of audio filepaths.
 88
 89        Generates the audios in the dataset by taking in a list of audio filepaths
 90        or a list of Audios
 91
 92        Args:
 93            v: Input for audios attribute that we're validating by generating the Audios if filepaths
 94                are provided or just the list of Audios if pre-generated and passed in
 95
 96        Returns:
 97            List of Audios that instantiates the audios attribute in the dataset
 98        """
 99        audio_list = []
100        if len(v) == 0:
101            return []
102        else:
103            for audio in v:
104                if isinstance(audio, Audio):
105                    audio_list.append(audio)
106                else:
107                    audio_list.append(Audio.from_filepath(audio))
108        return audio_list

151    def create_audio_split_for_pydra_task(self, batch_size: int = 1) -> List[List[Audio]]:
152        """Splits the audio data for Pydra tasks.
153
154        Creates a split of the audio data that can be used for creating individual Pydra tasks using
155        the .split functionality. Splits the data such that the inputs for a Pydra workflow are either
156        optimized for the GPU's batch size or a single Audio per CPU thread.
157
158        Args:
159            batch_size: How to batch Audios for a Pydra task; defaults to 1 since CPU won't batch
160
161        Returns:
162            List of Lists of Audio where each List of Audios will be an input to a Pydra task.
163            Each of the sublists are either of size 1 for CPUs or at most batch_size for GPU optimization.
164
165        Raises:
166            ValueError if the batch size is invalid (less than 1)
167        """
168        if batch_size > 1:
169            # Creates batches of at most size batch_size except the last which contains the remainder of audios
170            return [
171                self.audios[batch_size * i : min(batch_size * (i + 1), len(self.audios))]
172                for i in range(math.ceil(len(self.audios) / batch_size))
173            ]
174        elif batch_size < 1:
175            raise ValueError("Batch size must be greater than or equal to 1")
176        else:
177            return [[audio] for audio in self.audios]

246    def _get_dict_representation(self) -> Dict:
247        """Internal function for generating a dictionary representation of the dataset.
248
249        Returns:
250            Generates a dictionary representation of the dataset where the keys are participants,
251              sessions, audios, videos, and metadata.
252        """
253        audio_data: Dict[str, List] = {}
254        video_data: Dict[str, List] = {}
255        senselab_dict: Dict[str, Union[Dict[str, List], List]] = {
256            "participants": [],
257            "sessions": [],
258            "audios": audio_data,
259            "videos": video_data,
260            "metadata": self.metadata.copy(),
261        }
262        participants_data = []
263        sessions_data = []
264
265        video_frames_data = []
266        video_fps_data = []
267        video_path_data = []
268        video_metadata = []
269        video_audio_data = []
270        video_audio_metadata = []
271
272        audio_waveform_data = []
273        audio_metadata = []
274
275        for participant in self.get_participants():
276            participants_data.append({"id": participant.id, "metadata": participant.metadata.copy()})
277        senselab_dict["participants"] = participants_data
278
279        for session in self.get_sessions():
280            sessions_data.append({"id": session.id, "metadata": session.metadata.copy()})
281        senselab_dict["sessions"] = sessions_data
282
283        for audio in self.audios:
284            audio_waveform_data.append(
285                {
286                    "array": audio.waveform.T,
287                    "sampling_rate": audio.sampling_rate,
288                    "path": audio.generate_path(),
289                }
290            )
291            audio_metadata.append(audio.metadata.copy())
292        audio_data["audio"] = audio_waveform_data
293        audio_data["metadata"] = audio_metadata
294
295        for video in self.videos:
296            video_frames_data.append({"image": [to_pil_image(frame.numpy()) for frame in list(video.frames)]})
297            video_fps_data.append(video.frame_rate)
298            video_path_data.append(video.generate_path())
299            video_metadata.append(video.metadata.copy())
300            video_audio_data.append(
301                None
302                if not video.audio
303                else {
304                    "array": video.audio.waveform.T.to(torch.float32).numpy(),
305                    "sampling_rate": video.audio.sampling_rate,
306                    "path": video.audio.generate_path(),
307                }
308            )
309            video_audio_metadata.append(None if not video.audio else video.audio.metadata.copy())
310
311        video_data["frames"] = video_frames_data
312        video_data["frame_rate"] = video_fps_data
313        video_data["path"] = video_path_data
314        video_data["metadata"] = video_metadata
315        video_data["audio"] = video_audio_data
316        video_data["audio_metadata"] = video_audio_metadata
317        # raise ValueError('fuck')
318        return senselab_dict

320    def convert_senselab_dataset_to_hf_datasets(self) -> Dict[str, Dataset]:
321        """Converts Senselab datasets into HuggingFace datasets.
322
323        Returns:
324            A dictionary of HuggingFace datasets that represent the underlying Senselab dataset.
325              Currently only supports creating HuggingFace datasets for the Audio(s) and Video(s) in
326              the SenselabDataset. Videos in HuggingFace are not natively supported, so they are treated
327              as Sequences of images with a frame_rate.
328        """
329        senselab_dict = self._get_dict_representation()
330
331        # print(senselab_dict['videos']['audio'][0])
332
333        features = Features(
334            {
335                "frames": {"image": Sequence(feature=Image())},
336                "frame_rate": Value("float32"),
337                "path": Value("string"),
338                "metadata": {},
339                "audio": HFAudio(mono=False, sampling_rate=48000),
340                "audio_metadata": Value("string"),
341            }
342        )
343
344        audio_dataset = Dataset.from_dict(senselab_dict["audios"]).cast_column("audio", HFAudio(mono=False))
345
346        video_dataset = Dataset.from_dict(senselab_dict["videos"], features=features)
347        # print(video_dataset[0])
348
349        hf_datasets = {}
350        hf_datasets["audios"] = audio_dataset
351        hf_datasets["videos"] = video_dataset
352
353        # TODO: Create datasets for participants and sessions
354        return hf_datasets

356    @classmethod
357    def convert_hf_dataset_to_senselab_dataset(
358        cls, hf_datasets: Dict[str, Dataset], metadata: Dict = {}, transfer_metadata: bool = False
359    ) -> "SenselabDataset":
360        """Converts HuggingFace dataset to a Senselab dataset.
361
362        Convert HuggingFace dataset(s) to a SenselabDataset where each component of a SenselabDataset
363          (e.g. audios, videos, sessions, participants) are stored under different keys in the provided dictionary.
364
365        Args:
366            hf_datasets: Dictionary of the different individual components that make up a SenselabDataset. Audios must
367              be organized as HuggingFace Audio(s) which is a dictionary with an array attribute, sampling rate, and
368              a path. Videos are not natively supported by HuggingFace, so SenselabDataset expects a sequence of
369              frames that are each an image, a frame_rate argument, and optionally the associated audio with the
370              video.
371            metadata: Dictionary of additional dataset level metadata. Differs from metadata inside each HuggingFace
372              dataset that is provided and is unaffected by transder_metadata.
373            transfer_metadata: Specifies whether to generate metadata from extraneous attributes in the
374              HuggingDataset in addition to any specified ina metadata field, otherwise they are ignored.
375              Defaults to False.
376
377        Returns:
378            The generated SenselabDataset from the provided fields in the hf_datasets dictionary. Currently does not
379              support converting sessions or participants.
380        """
381        audios = []
382        videos = []
383        sessions: Dict[str, Session] = {}
384        participants: Dict[str, Participant] = {}
385
386        if "audios" in hf_datasets:
387            audio_dataset = hf_datasets["audios"]
388            for audio in audio_dataset:  # Equivalent of running over each row in the Dataset
389                audio_metadata = audio["metadata"] if "metadata" in audio else {}
390                if transfer_metadata:
391                    for feature in audio_dataset.features:
392                        if feature == "metadata" or feature == "audio":
393                            continue
394                        audio_metadata[feature] = audio[feature]
395                audios.append(
396                    Audio(
397                        waveform=audio["audio"]["array"],
398                        sampling_rate=audio["audio"]["sampling_rate"],
399                        orig_path_or_id=audio["audio"]["path"],
400                        metadata=audio_metadata,
401                    )
402                )
403
404        if "videos" in hf_datasets:
405            video_dataset = hf_datasets["videos"]
406            for video in video_dataset:
407                video_metadata = video["metadata"] if "metadata" in video else {}
408                if transfer_metadata:
409                    for feature in video_dataset.features:
410                        if (
411                            feature == "metadata"
412                            or feature == "frames"
413                            or feature == "frame_rate"
414                            or feature == "path"
415                            or feature == "audio"
416                        ):
417                            continue
418                        video_metadata[feature] = video[feature]
419                videos.append(
420                    Video(
421                        frames=video["frames"]["image"],
422                        frame_rate=video["frame_rate"],
423                        metadata=video_metadata,
424                        orig_path_or_id=video["path"],
425                        audio=Audio(  # Assumes audio metadata is stored a level higher within the video's metadata
426                            waveform=video["audio"]["array"],
427                            sampling_rate=video["audio"]["sampling_rate"],
428                            orig_path_or_id=video["audio"]["path"],
429                        )
430                        if video["audio"]
431                        else None,
432                    )
433                )
434        if "sessions" in hf_datasets:
435            pass
436        if "participants" in hf_datasets:
437            pass
438
439        return SenselabDataset(
440            participants=participants, sessions=sessions, audios=audios, videos=videos, metadata=metadata.copy()
441        )
442
443    def __eq__(self, other: object) -> bool:
444        """Overloads the default BaseModel equality to correctly check that datasets are equivalent."""
445        if isinstance(other, SenselabDataset):
446            return (
447                self.audios == other.audios
448                and self.videos == other.videos
449                and self.participants == other.participants
450                and self.sessions == other.sessions
451            )
452        return False

364          (e.g. audios, videos, sessions, participants) are stored under different keys in the provided dictionary.
365
366        Args:
367            hf_datasets: Dictionary of the different individual components that make up a SenselabDataset. Audios must
368              be organized as HuggingFace Audio(s) which is a dictionary with an array attribute, sampling rate, and

370              frames that are each an image, a frame_rate argument, and optionally the associated audio with the

385        participants: Dict[str, Participant] = {}

388            audio_dataset = hf_datasets["audios"]
389            for audio in audio_dataset:  # Equivalent of running over each row in the Dataset
390                audio_metadata = audio["metadata"] if "metadata" in audio else {}
391                if transfer_metadata:
392                    for feature in audio_dataset.features:
393                        if feature == "metadata" or feature == "audio":
394                            continue
395                        audio_metadata[feature] = audio[feature]
396                audios.append(
397                    Audio(
398                        waveform=audio["audio"]["array"],
399                        sampling_rate=audio["audio"]["sampling_rate"],
400                        orig_path_or_id=audio["audio"]["path"],
401                        metadata=audio_metadata,
402                    )
403                )

407            for video in video_dataset:
408                video_metadata = video["metadata"] if "metadata" in video else {}
409                if transfer_metadata:

411                        if (
412                            feature == "metadata"
413                            or feature == "frames"

431                        if video["audio"]

437        if "participants" in hf_datasets:
438            pass

Converts HuggingFace dataset to a Senselab dataset.

Convert HuggingFace dataset(s) to a SenselabDataset where each component of a SenselabDataset (e.g. audios, videos, sessions, participants) are stored under different keys in the provided dictionary.

Arguments:
  • hf_datasets: Dictionary of the different individual components that make up a SenselabDataset. Audios must be organized as HuggingFace Audio(s) which is a dictionary with an array attribute, sampling rate, and a path. Videos are not natively supported by HuggingFace, so SenselabDataset expects a sequence of frames that are each an image, a frame_rate argument, and optionally the associated audio with the video.
  • metadata: Dictionary of additional dataset level metadata. Differs from metadata inside each HuggingFace dataset that is provided and is unaffected by transder_metadata.
  • transfer_metadata: Specifies whether to generate metadata from extraneous attributes in the HuggingDataset in addition to any specified ina metadata field, otherwise they are ignored. Defaults to False.
Returns:

The generated SenselabDataset from the provided fields in the hf_datasets dictionary. Currently does not support converting sessions or participants.
