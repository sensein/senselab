 93class MediaPipePoseEstimator(PoseEstimator):
 94    """MediaPipe implementation of pose estimation.
 95
 96    Attributes:
 97        model_path (str): Path to the MediaPipe model file.
 98        num_individuals (int): Number of individuals to detect.
 99    """
100
101    def __init__(self, model_type: str = "lite") -> None:
102        """Initialize the MediaPipePoseEstimator.
103
104        Args:
105            model_type (str): Type of MediaPipe model to use ('lite', 'full', 'heavy').
106        """
107        if not MEDIAPIPE_AVAILABLE:
108            raise ModuleNotFoundError(
109                "`mediapipe` is not installed. "
110                "Please install senselab video dependencies using `pip install 'senselab[video]'`."
111            )
112
113        self.model_path = get_model("mediapipe", model_type)
114        self.num_individuals = 1
115        self._detector = None
116
117    def detector(self, num_individuals: int = 1) -> "vision.PoseLandmarker":
118        """Initialization of the MediaPipe detector.
119
120        Args:
121            num_individuals (int): Maximum number of individuals to detect. Defaults to 1.
122
123        Returns:
124            vision.PoseLandmarker: A MediaPipe pose landmarker object.
125        """
126        if self._detector is None or self.num_individuals != num_individuals:
127            base_options = python.BaseOptions(model_asset_path=self.model_path)
128            options = vision.PoseLandmarkerOptions(
129                base_options=base_options, output_segmentation_masks=True, num_poses=num_individuals
130            )
131            self.num_individuals = num_individuals
132            self._detector = vision.PoseLandmarker.create_from_options(options)
133        return self._detector
134
135    def estimate(self, image: np.ndarray, num_individuals: int = 1) -> ImagePose:
136        """Estimate poses using MediaPipe.
137
138        Args:
139            image (np.ndarray): Input image in RGB format.
140            num_individuals (int): Maximum number of individuals to detect. Defaults to 1.
141
142        Returns:
143            ImagePose: An object containing detected poses and metadata.
144
145        Raises:
146            ValueError: If `num_individuals` is not a positive integer.
147        """
148        if not isinstance(num_individuals, int) or num_individuals < 0:
149            raise ValueError("Number of individuals must be an integer >=0")
150
151        mp_image = mp.Image(image_format=mp.ImageFormat.SRGB, data=image)
152
153        detection_result = self.detector(num_individuals).detect(mp_image)
154
155        individuals = self.extract_landmarks(detection_result)
156        return ImagePose(
157            image=image,
158            individuals=individuals,
159            model=PoseModel.MEDIAPIPE,
160        )
161
162    def estimate_from_path(self, image_path: str, num_individuals: int = 1) -> ImagePose:
163        """Estimate poses in image from file path using MediaPipe.
164
165        Args:
166            image_path (str): Path to the input image file.
167            num_individuals (int): Maximum number of individuals to detect. Defaults to 1.
168
169        Returns:
170            ImagePose: An object containing detected poses and metadata.
171
172        Raises:
173            FileNotFoundError: If the specified image file does not exist.
174            ValueError: If `num_individuals` is not a positive integer.
175        """
176        if not CV2_AVAILABLE:
177            raise ModuleNotFoundError(
178                "`opencv-python` is not installed. "
179                "Please install senselab video dependencies using `pip install 'senselab[video]'`."
180            )
181
182        if not os.path.exists(image_path):
183            raise FileNotFoundError(f"Image not found at: {image_path}")
184
185        # Load and process the image
186        image = cv2.imread(image_path)
187        image = cv2.cvtColor(image, cv2.COLOR_BGR2RGB)
188        return self.estimate(image, num_individuals)
189
190    def extract_landmarks(self, detection_result: "vision.PoseLandmarkerResult") -> List[IndividualPose]:
191        """Extract normalized and world landmarks from the detection result.
192
193        Args:
194            detection_result (vision.PoseLandmarkerResult): The result from the MediaPipe pose detector.
195
196        Returns:
197            List[IndividualPose]: A list of `IndividualPose` objects representing detected individuals.
198        """
199        individuals = []
200        for idx, (norm_landmarks, world_landmarks) in enumerate(
201            zip(detection_result.pose_landmarks, detection_result.pose_world_landmarks)
202        ):
203            norm_dict = {
204                MEDIAPIPE_KEYPOINT_MAPPING.get(i, f"keypoint_{i}"): MediaPipePoseLandmark(
205                    x=lm.x, y=lm.y, z=lm.z, visibility=lm.visibility
206                )
207                for i, lm in enumerate(norm_landmarks)
208            }
209
210            world_dict = {
211                MEDIAPIPE_KEYPOINT_MAPPING.get(i, f"keypoint_{i}"): MediaPipePoseLandmark(
212                    x=lm.x, y=lm.y, z=lm.z, visibility=norm_landmarks[i].visibility
213                )
214                for i, lm in enumerate(world_landmarks)
215            }
216
217            individual = IndividualPose(
218                individual_index=idx,
219                normalized_landmarks=norm_dict,
220                world_landmarks=world_dict,
221            )
222            individuals.append(individual)
223        return individuals

226class YOLOPoseEstimator(PoseEstimator):
227    """YOLO implementation of pose estimation.
228
229    Attributes:
230        model_path (str): Path to the YOLO model file.
231    """
232
233    def __init__(self, model_type: str = "8n") -> None:
234        """Initialize the YOLOPoseEstimator.
235
236        Args:
237            model_type (str): Type of YOLO model to use (e.g. '8n', '11p', '11s').
238        """
239        if not YOLO_AVAILABLE:
240            raise ModuleNotFoundError(
241                "`ultralytics` is not installed. "
242                "Please install senselab video dependencies using `pip install 'senselab[video]'`."
243            )
244
245        self.model_path = get_model("yolo", model_type)
246        self._model = YOLO(self.model_path)
247
248    def estimate(self, image: np.ndarray) -> ImagePose:
249        """Estimate poses using YOLO.
250
251        Args:
252            image (np.ndarray): Input image in RGB format.
253
254        Returns:
255            ImagePose: An object containing detected poses and metadata.
256        """
257        results = self._model(image, verbose=False)
258
259        if results[0].keypoints is None or results[0].keypoints.data.numel() == 0:
260            # No individuals detected
261            return ImagePose(
262                image=image,
263                individuals=[],
264                model=PoseModel.YOLO,
265            )
266
267        individuals = []
268        for idx, person_keypoints in enumerate(results[0].keypoints):
269            confidence_values = person_keypoints.conf.squeeze()  # Extract confidence values
270
271            normalized_dict = {
272                YOLO_KEYPOINT_MAPPING.get(i, f"keypoint_{i}"): YOLOPoseLandmark(
273                    x=kp[0].item(), y=kp[1].item(), confidence=confidence_values[i].item()
274                )
275                for i, kp in enumerate(person_keypoints.xyn[0])
276            }
277
278            individual = IndividualPose(
279                individual_index=idx,
280                normalized_landmarks=normalized_dict,
281            )
282            individuals.append(individual)
283
284        return ImagePose(
285            image=image,
286            individuals=individuals,
287            model=PoseModel.YOLO,
288        )

291        """Estimate poses in image from file path using YOLO.

313        image = cv2.cvtColor(image, cv2.COLOR_BGR2RGB)
314        return self.estimate(image)
